1   package com.epam.ships.server.gamestates;
2   
3   import com.epam.ships.infra.communication.api.message.Header;
4   import com.epam.ships.infra.logging.api.Target;
5   import com.epam.ships.infra.logging.core.SharedLogger;
6   import com.epam.ships.server.CommunicationBus;
7   import com.epam.ships.server.MessageSender;
8   import com.epam.ships.server.TurnManager;
9   
10  /**
11   * Represents game state where game have a winner.
12   *
13   * @author Piotr Czy≈º
14   * @since 2018-01-02
15   */
16  public class GameEndWithWinState implements GameState {
17    private final MessageSender messageSender;
18    private final Target logger = new SharedLogger(GameEndWithWinState.class);
19    private CommunicationBus communicationBus;
20    private TurnManager turnManager;
21  
22    /**
23     * Its create state of game when we have winner.
24     *
25     * @param communicationBus client server communication bus
26     * @param turnManager      take care about players turns
27     */
28    public GameEndWithWinState(final CommunicationBus communicationBus,
29                               final TurnManager turnManager) {
30      this.communicationBus = communicationBus;
31      this.turnManager = turnManager;
32      messageSender = new MessageSender(communicationBus, logger);
33    }
34  
35    /**
36     * Process end of game with the winner, and close communication bus.
37     *
38     * @return GameState
39     */
40    @Override
41    public GameState process() {
42      handleEndOfGame();
43      communicationBus.stop();
44      logger.info("Game ended");
45      return this;
46    }
47  
48    /**
49     * Set that game should not by continued since we have a winner.
50     *
51     * @return boolean
52     */
53    @Override
54    public boolean shouldBeContinued() {
55      return false;
56    }
57  
58    private void handleEndOfGame() {
59      messageSender.send(turnManager.getCurrentPlayer(), Header.WIN);
60      messageSender.send(turnManager.getOtherPlayer(), Header.LOSE);
61    }
62  }
