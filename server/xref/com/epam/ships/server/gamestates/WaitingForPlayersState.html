9   /**
10   * Represents state where player is waiting for opponent.
11   *
12   * @author Piotr Czy≈º
13   * @since 2018-01-02
14   */
15  public class WaitingForPlayersState implements GameState {
16    private final Target logger = new SharedLogger(WaitingForPlayersState.class);
17    private CommunicationBus communicationBus;
18  
19    /**
20     * Create instance of WaitingForPlayersState.
21     *
22     * @param communicationBus client server communication bus
23     */
24    public WaitingForPlayersState(final CommunicationBus communicationBus) {
25      this.communicationBus = communicationBus;
26    }
27  
28    /**
29     * Process game from waiting for opponent to FleetPlacementState.
30     *
31     * @return GameState - FleetPlacementState
32     */
33    @Override
34    public GameState process() {
35      communicationBus.start();
36  
37      MessageSender messageSender = new MessageSender(communicationBus, logger);
38      messageSender.sendToAll(Header.OPPONENT_CONNECTED);
39  
40      return new FleetPlacementState(communicationBus);
41    }
42  }
