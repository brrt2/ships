3   import pl.korotkevics.ships.server.communication.CommunicationBus;
4   import pl.korotkevics.ships.server.game.gamestates.GameState;
5   import pl.korotkevics.ships.shared.infra.logging.api.Target;
6   import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;

8   /**
9    * Represents game state in which one of players quits in the middle of game.
10   *
11   * @author Piotr Czy≈º
12   * @since 2018-01-02
13   */
14  public class GameEndWithWalkoverState implements GameState {
15    private final Target logger = new SharedLogger(GameEndWithWalkoverState.class);
16    private final CommunicationBus communicationBus;
17  
18    /**
19     * It creates state in which there is a walk-over.
20     *
21     * @param communicationBus client server communication bus
22     */
23    public GameEndWithWalkoverState(final CommunicationBus communicationBus) {
24      this.communicationBus = communicationBus;
25    }
26  
27    /**
28     * Process game end with walk-over.
29     *
30     * @return return a state in which game will be discontinued
31     * @see #shouldBeContinued()
32     */
33    @Override
34    public GameState process() {
35      logger.info("Game ends - walk-over");
36      communicationBus.stop();
37      return this;
38    }
39  
40    /**
41     * Set that game should not by continued since there is no opponent.
42     *
43     * @return false when game can not be continued because there is no opponent.
44     */
45    @Override
46    public boolean shouldBeContinued() {
47      return false;
48    }
49  }
