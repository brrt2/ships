001package pl.korotkevics.ships.server.communication;
002
003import pl.korotkevics.ships.shared.infra.logging.api.Target;
004import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;
005
006import java.io.IOException;
007import java.net.ServerSocket;
008import java.net.Socket;
009import java.util.ArrayList;
010import java.util.List;
011
012/**
013 * It starts a server socket and accepts two client sockets.
014 *
015 * @author Piotr Czy≈º, Magdalena Aarsman, Sandor Korotkevics
016 * @since 2017-12-09
017 */
018public class AppServer {
019
020  private final Target logger = new SharedLogger(AppServer.class);
021
022  private final ServerSocket serverSocket;
023
024  public AppServer(int port) throws IOException {
025    logger.info("Server is up and waiting for clients..");
026    this.serverSocket = new ServerSocket(port);
027  }
028
029  /**
030   * It connects two clients.
031   */
032  public List<Socket> connectClients() {
033    List<Socket> clients = new ArrayList<>();
034    logger.info("Waiting for the 1st client.. ");
035    acceptClient(clients);
036    logger.info("1st client connected... ");
037    logger.info("waiting for the 2nd client..");
038    acceptClient(clients);
039    logger.info("2nd client connected... ");
040    logger.info("Clients are connected");
041    return clients;
042  }
043
044  /**
045   * It accepts a client socket
046   * while storing it in a list.
047   */
048  private void acceptClient(List<Socket> clients) {
049    try {
050      Socket client = serverSocket.accept();
051      clients.add(client);
052    } catch (IOException e) {
053      logger.error(e.getMessage());
054    }
055  }
056
057}
