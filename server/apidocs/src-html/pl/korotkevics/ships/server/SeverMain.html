001package pl.korotkevics.ships.server;
002
003import pl.korotkevics.ships.server.communication.AppServer;
004import pl.korotkevics.ships.server.communication.CommunicationBus;
005import pl.korotkevics.ships.server.game.Game;
006import pl.korotkevics.ships.server.game.gamestates.GameState;
007import pl.korotkevics.ships.server.game.gamestates.connection.WaitingForPlayersState;
008
009import java.io.IOException;
010import java.net.Socket;
011import java.util.List;
012
013/**
014 * Main class of server application.
015 *
016 * @author Piotr Czy≈º
017 * @since 2017-12-09
018 */
019public class SeverMain {
020  private static final int SERVER_PORT = 8189;
021  /**
022   * Main server method. It creates communication bus and starts the game.
023   *
024   * @param args is a conventional main method parameter.
025   * @throws IOException can be thrown due to establishing a socket connection.
026   */
027  public static void main(String[] args) throws IOException {
028    AppServer appServer = new AppServer(SERVER_PORT);
029    int gameId = 1;
030    /*
031    Sonar reports an infinitive loop as a critical bug, and therefore
032    we could not think of a better workaround than presenting it as a
033    an effective infinitive loop (with true extracted to a flag).
034     */
035    boolean shouldRun = true;
036    while (shouldRun) {
037      CommunicationBus communicationBus = new CommunicationBus(appServer.connectClients());
038      final int finalGameId = gameId;
039      new Thread(() -> {
040        Thread.currentThread().setName("GAME: " + finalGameId);
041        GameState initialState = new WaitingForPlayersState(communicationBus);
042        new Game(initialState).play();
043      }).start();
044      gameId++;
045    }
046  }
047}
