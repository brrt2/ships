001package com.epam.ships.server.gamestates;
002
003import com.epam.ships.shared.infra.logging.api.Target;
004import com.epam.ships.shared.infra.logging.core.SharedLogger;
005import com.epam.ships.server.CommunicationBus;
006
007/**
008 * Represents game state where one of players quit in the middle of game.
009 *
010 * @author Piotr Czy≈º
011 * @since 2018-01-02
012 */
013public class GameEndWithWalkoverState implements GameState {
014  private final Target logger = new SharedLogger(GameEndWithWalkoverState.class);
015  private final CommunicationBus communicationBus;
016
017  /**
018   * It creates state in which there is a walk-over.
019   *
020   * @param communicationBus client server communication bus
021   */
022  public GameEndWithWalkoverState(final CommunicationBus communicationBus) {
023    this.communicationBus = communicationBus;
024  }
025
026  /**
027   * Process game end with walk-over.
028   *
029   * @return return a state in which game will be discontinued
030   * @see #shouldBeContinued()
031   */
032  @Override
033  public GameState process() {
034    logger.info("Game ends - walk-over");
035    communicationBus.stop();
036    return this;
037  }
038
039  /**
040   * Set that game should not by continued since there is no opponent.
041   *
042   * @return false when game can not be continued because there is no opponent.
043   */
044  @Override
045  public boolean shouldBeContinued() {
046    return false;
047  }
048}
