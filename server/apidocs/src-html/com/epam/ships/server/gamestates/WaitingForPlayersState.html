001package com.epam.ships.server.gamestates;
002
003import com.epam.ships.shared.infra.communication.api.message.Header;
004import com.epam.ships.shared.infra.logging.api.Target;
005import com.epam.ships.shared.infra.logging.core.SharedLogger;
006import com.epam.ships.server.CommunicationBus;
007import com.epam.ships.server.MessageSender;
008
009/**
010 * Represents state where player is waiting for opponent.
011 *
012 * @author Piotr Czy≈º
013 * @since 2018-01-02
014 */
015public class WaitingForPlayersState implements GameState {
016  private final Target logger = new SharedLogger(WaitingForPlayersState.class);
017  private CommunicationBus communicationBus;
018
019  /**
020   * Create instance of WaitingForPlayersState.
021   *
022   * @param communicationBus client server communication bus
023   */
024  public WaitingForPlayersState(final CommunicationBus communicationBus) {
025    this.communicationBus = communicationBus;
026  }
027
028  /**
029   * Process game from waiting for opponent to FleetPlacementState.
030   *
031   * @return GameState - FleetPlacementState
032   */
033  @Override
034  public GameState process() {
035    communicationBus.start();
036
037    MessageSender messageSender = new MessageSender(communicationBus, logger);
038    messageSender.sendToAll(Header.OPPONENT_CONNECTED);
039
040    return new FleetPlacementState(communicationBus);
041  }
042}
