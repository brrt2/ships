001package pl.korotkevics.ships.server;
002
003import org.testng.annotations.Test;
004
005import java.io.IOException;
006import java.net.Socket;
007
008import static org.testng.Assert.assertEquals;
009
010@Test(groups = {"integration"})
011public class AppServerTest {
012
013  public void shouldConnectTwoClients() throws IOException, InterruptedException {
014    //given
015    AppServer appServer = new AppServer(8989);
016
017    //when
018    connectClient();
019    connectClient();
020    appServer.connectClients();
021
022    //then
023    assertEquals(appServer.getClientSockets().size(), 2);
024  }
025
026  private void connectClient() throws IOException, InterruptedException {
027    new Thread(() -> {
028      try {
029        Thread.sleep(10);
030        new Socket("127.0.0.1", 8989);
031      } catch (final InterruptedException e) {
032        e.printStackTrace();
033      } catch (final IOException e) {
034        e.printStackTrace();
035      }
036    }).start();
037  }
038}
