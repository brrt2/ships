001package pl.korotkevics.ships.server.gamestates.fleetplacement;
002
003import org.testng.annotations.Test;
004import pl.korotkevics.ships.server.CommunicationBus;
005import pl.korotkevics.ships.server.WrappedClient;
006import pl.korotkevics.ships.server.gamestates.GameState;
007import pl.korotkevics.ships.server.gamestates.play.PlayState;
008import pl.korotkevics.ships.shared.fleet.Fleet;
009import pl.korotkevics.ships.shared.fleet.Mast;
010import pl.korotkevics.ships.shared.fleet.Ship;
011import pl.korotkevics.ships.shared.infra.communication.api.Message;
012import pl.korotkevics.ships.shared.infra.communication.core.message.MessageBuilder;
013
014import java.util.Arrays;
015
016import static org.mockito.Mockito.mock;
017import static org.mockito.Mockito.when;
018import static org.testng.Assert.assertEquals;
019import static org.testng.Assert.assertNotNull;
020
021@Test
022public class FleetPlacementStateTest {
023
024  public void shouldProcessState() {
025    //given
026    CommunicationBus communicationBus = mock(CommunicationBus.class);
027    WrappedClient player1 = mock(WrappedClient.class);
028    WrappedClient player2 = mock(WrappedClient.class);
029    when(communicationBus.getFirstClient()).thenReturn(player1);
030    when(communicationBus.getSecondClient()).thenReturn(player2);
031
032    Message fleetMsg = new MessageBuilder()
033        .withFleet(Fleet.ofShips(Arrays.asList(Ship.ofMasts(Mast.ofIndex("3"), Mast.ofIndex("2"), Mast.ofIndex("1")))))
034        .build();
035    when(communicationBus.receive(player1)).thenReturn(fleetMsg);
036    when(communicationBus.receive(player2)).thenReturn(fleetMsg);
037
038    //when
039    GameState playState = new FleetPlacementState(communicationBus).process();
040
041    //then
042    assertNotNull(playState);
043    assertEquals(PlayState.class, playState.getClass());
044  }
045}
