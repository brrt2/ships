001package pl.korotkevics.ships.server.gamestates.play;
002
003import org.testng.annotations.Test;
004import pl.korotkevics.ships.server.CommunicationBus;
005import pl.korotkevics.ships.server.MessageReceiver;
006import pl.korotkevics.ships.server.WrappedClient;
007import pl.korotkevics.ships.shared.infra.communication.api.Message;
008import pl.korotkevics.ships.shared.infra.communication.api.message.Author;
009import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
010import pl.korotkevics.ships.shared.infra.communication.core.message.MessageBuilder;
011
012import static org.mockito.Mockito.*;
013import static org.testng.Assert.assertFalse;
014import static org.testng.Assert.assertTrue;
015
016@Test
017public class MessageReceiverTest {
018  public void shouldReceiveMessageFromPlayer() {
019    //given
020    CommunicationBus communicationBus = mock(CommunicationBus.class);
021    WrappedClient wrappedClient = mock(WrappedClient.class);
022
023    //when
024    MessageReceiver messageReceiver = new MessageReceiver(communicationBus);
025    messageReceiver.receive(wrappedClient);
026
027    //then
028    verify(communicationBus, times(1)).receive(wrappedClient);
029  }
030
031  public void shouldBeAShotIfLastMessageWasShot() {
032    //given
033    Message lastMessage = new MessageBuilder()
034        .withAuthor(Author.SERVER)
035        .withHeader(Header.SHOT)
036        .build();
037    CommunicationBus communicationBus = mock(CommunicationBus.class);
038    WrappedClient wrappedClient = mock(WrappedClient.class);
039    when(communicationBus.receive(wrappedClient)).thenReturn(lastMessage);
040
041    MessageReceiver messageReceiver = new MessageReceiver(communicationBus);
042    messageReceiver.receive(wrappedClient);
043
044    //when
045    boolean result = messageReceiver.isAShot();
046
047    //then
048    assertTrue(result);
049  }
050
051  public void shouldNotBeAShotIfLastMessageWasNotShot() {
052    //given
053    Message lastMessage = new MessageBuilder()
054        .withAuthor(Author.AUTO)
055        .withHeader(Header.CONNECTION)
056        .build();
057    CommunicationBus communicationBus = mock(CommunicationBus.class);
058    WrappedClient wrappedClient = mock(WrappedClient.class);
059    when(communicationBus.receive(wrappedClient)).thenReturn(lastMessage);
060
061    MessageReceiver messageReceiver = new MessageReceiver(communicationBus);
062    messageReceiver.receive(wrappedClient);
063    //when
064    boolean result = messageReceiver.isAShot();
065
066    //then
067    assertFalse(result);
068  }
069}
