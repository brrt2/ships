1   package com.epam.ships.server;
2   
3   import com.epam.ships.infra.communication.api.Message;
4   import com.epam.ships.infra.communication.api.message.Author;
5   import com.epam.ships.infra.communication.api.message.Header;
6   import com.epam.ships.infra.communication.core.message.MessageBuilder;
7   
8   import java.util.EnumMap;
9   
10  /**
11   * Repository of message that server can send.
12   *
13   * @author Piotr Czy≈º
14   * @since 2018-01-02
15   */
16  public class MessageRepository {
17  
18    private EnumMap<Header, Message> messages;
19  
20    MessageRepository() {
21      this.messages = new EnumMap<>(Header.class);
22      messages.put(Header.OPPONENT_CONNECTED, this.opponentConnectedMessage());
23      messages.put(Header.WIN, this.winMessage());
24      messages.put(Header.LOSE, this.loseMessage());
25      messages.put(Header.PLACEMENT, this.askForFleetMessage());
26      messages.put(Header.HIT, this.hitMessage());
27      messages.put(Header.MISS, this.missMessage());
28      messages.put(Header.SHIP_DESTRUCTED, this.shipDestructedMessage());
29      messages.put(Header.YOUR_TURN, this.yourTurnMessage());
30    }
31  
32    /**
33     * Gets message from repository based on message header.
34     *
35     * @param header header of wonted message
36     * @return proper Message from MessageRepository
37     */
38    public Message getMessage(final Header header) {
39      return messages.get(header);
40    }
41  
42    private Message yourTurnMessage() {
43      return new MessageBuilder()
44          .withAuthor(Author.SERVER)
45          .withHeader(Header.YOUR_TURN)
46          .build();
47    }
48  
49    private Message shipDestructedMessage() {
50      return new MessageBuilder()
51          .withAuthor(Author.SERVER)
52          .withHeader(Header.SHIP_DESTRUCTED)
53          .build();
54    }
55  
56    private Message missMessage() {
57      return new MessageBuilder()
58          .withAuthor(Author.SERVER)
59          .withHeader(Header.MISS)
60          .build();
61    }
62  
63    private Message hitMessage() {
64      return new MessageBuilder()
65          .withAuthor(Author.SERVER)
66          .withHeader(Header.HIT)
67          .build();
68    }
69  
70    private Message askForFleetMessage() {
71      return new MessageBuilder()
72          .withAuthor(Author.SERVER)
73          .withHeader(Header.PLACEMENT)
74          .build();
75    }
76  
77    private Message loseMessage() {
78      return new MessageBuilder()
79          .withAuthor(Author.SERVER)
80          .withHeader(Header.LOSE)
81          .build();
82    }
83  
84    private Message winMessage() {
85      return new MessageBuilder()
86          .withAuthor(Author.SERVER)
87          .withHeader(Header.WIN)
88          .build();
89    }
90  
91    private Message opponentConnectedMessage() {
92      return new MessageBuilder()
93          .withAuthor(Author.SERVER)
94          .withHeader(Header.OPPONENT_CONNECTED)
95          .withStatement("true")
96          .build();
97    }
98  }
