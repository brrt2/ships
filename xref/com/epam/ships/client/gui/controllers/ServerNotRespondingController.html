20  public class ServerNotRespondingController {
21  
22    private static final Target logger = new SharedLogger(Client.class);
23  
24    @FXML
25    private AnchorPane mainAnchorPane;
26  
27    @FXML
28    private void onTryAgainPressed() {
29      try {
30        final String connectWindowUrl = "/fxml/connectWindow.fxml";
31        final FXMLLoader connectLoader = new FXMLLoader(getClass().getResource(connectWindowUrl));
32        final Parent connectWindow = connectLoader.load();
33        final Pane mainPane = (Pane) mainAnchorPane.getParent();
34        mainPane.getChildren().clear();
35        mainPane.getChildren().setAll(connectWindow);
36      } catch (IOException e) {
37        logger.error(e.getMessage());
38      }
39    }
40  }
