3   /**
4    * Enables validation of a port number.
5    * @author Magdalena Aarsman
6    * @since 2017-12-16
7    */
8   public class PortValidator {
9   
10    /**
11     * Validate the port number, returns port as Integer when it is valid.
12     * @param providedPort - port provided by user as a String
13     * @return - port number
14     * @throws IllegalArgumentException if port is invalid.
15     */
16    public int asInt(String providedPort) {
17      int port = Integer.parseInt(providedPort);
18      checkRange(port);
19      return port;
20    }
21  
22    private void checkRange(int port) {
23      final int maxValidPort = 0xFFFF;
24      final int minValidPort = 1024;
25      if (port <= minValidPort || port > maxValidPort) {
26        throw new IllegalArgumentException("port value is out of range");
27      }
28    }
29  }
