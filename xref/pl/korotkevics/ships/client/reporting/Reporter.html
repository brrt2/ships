20  /**
21   * @author Sandor Korotkevics
22   * @since 2018-01-24
23   * @see ReportingOption
24   *
25   * Reports game messages to a configurable target.
26   */
27  public class Reporter {
28    
29    private static final Target logger = new SharedLogger(Reporter.class);
30    
31    private static final String PATH_SEPARATOR = ".";
32    
33    private static final String TARGET = "target";
34    
35    private static final String OPTIONS = "options";
36    
37    private static final String DESTINATION = "destination";
38    
39    private static final String LOGGER_OPTION = "logger";
40    
41    private static final String SOCKET_OPTION = "socket";
42    
43    private static final String FILE_OPTION = "file";
44    
45    private static final String WINDOW_OPTION = "window";
46    
47    private static final String NAME = "name";
48    
49    private static final String HOST = "host";
50    
51    private static final String PORT = "port";
52    
53    private static final String TITLE = "title";
54    
55    private static final String HEIGHT = "height";
56    
57    private static final String WIDTH = "width";
58    
59    private final Config config;
60    
61    private Socket socket;
62    
63    /**
64     * Reports a game message to a configurable target.
65     * Target is configured with a .conf file.
66     *
67     * @param message from a game trigger to be reported.
68     */
69    public void report(final String message) {
70      try {
71        if (this.getCurrentDestination().equals(ReportingOption.FILE.toString())) {
72          this.writeToFile(message);
73        } else if (this.getCurrentDestination().equals(ReportingOption.SOCKET.toString())) {
74          this.establishConnection();
75          this.writeToSocket(message);
76        } else if (this.getCurrentDestination().equals(ReportingOption.LOGGER.toString())) {
77          this.log(message);
78        }
79      } catch (ConfigException e) {
80        logger.error(e.getMessage());
81      }
82    }
83    
84    private void log(final String message) {
85      logger.report(message, this.getDestinationLoggerName());
86    }
87    
88    private void writeToFile(final String message) {
89      try {
90        Files.write(Paths.get(this.getDestinationFileName()), StringUtils.defaultIfEmpty(message
91                                                                                             +
92                                                                                             System.lineSeparator(), StringUtils.EMPTY).getBytes(), StandardOpenOption.APPEND);
93      } catch (IOException e) {
94        logger.error(e.getMessage());
95      }
96    }
97    
98    private void writeToSocket(final String message) {
99      PrintWriter printWriter;
100     try {
101       printWriter = new PrintWriter(new OutputStreamWriter(this.socket.getOutputStream(), "UTF-8"), true);
102       printWriter.println(message);
103     } catch (IOException e) {
104       logger.error(e.getMessage());
105     }
106   }
107   
108   private void establishConnection() {
109     if (this.socket == null) {
110       try {
111         final InetAddress inetAddress = InetAddress.getByName(this.getDestinationHost());
112         this.socket = new Socket();
113         this.socket.connect(new InetSocketAddress(inetAddress, this.getDestinationPort()));
114       } catch (IOException e) {
115         logger.error(e.getMessage());
116       }
117     }
118   }
119   
120   public Reporter(final String fileName) {
121     this.config = ConfigFactory.load(fileName);
122   }
123   
124   String getCurrentDestination() {
125     return this.config.getString(DESTINATION + PATH_SEPARATOR + TARGET);
126   }
127   
128   String getDestinationFileName() {
129     return this.config.getString(this.getOptionsPath() + FILE_OPTION + PATH_SEPARATOR + NAME);
130   }
131   
132   String getDestinationHost() {
133     return this.config.getString(this.getOptionsPath() + SOCKET_OPTION + PATH_SEPARATOR + HOST);
134   }
135   
136   int getDestinationPort() {
137     return this.config.getInt(this.getOptionsPath() + SOCKET_OPTION + PATH_SEPARATOR + PORT);
138   }
139   
140   String getDestinationLoggerName() {
141     return this.config.getString(this.getOptionsPath() + LOGGER_OPTION + PATH_SEPARATOR + NAME);
142   }
143   
144   String getDestinationWindowTitle() {
145     return this.config.getString(this.getOptionsPath() + WINDOW_OPTION + PATH_SEPARATOR + TITLE);
146   }
147   
148   int getDestinationWindowHeight() {
149     return this.config.getInt(this.getOptionsPath() + WINDOW_OPTION + PATH_SEPARATOR + HEIGHT);
150   }
151   
152   int getDestinationWindowWidth() {
153     return this.config.getInt(this.getOptionsPath() + WINDOW_OPTION + PATH_SEPARATOR + WIDTH);
154   }
155   
156   private String getOptionsPath() {
157     return OPTIONS + PATH_SEPARATOR;
158   }
159 }
