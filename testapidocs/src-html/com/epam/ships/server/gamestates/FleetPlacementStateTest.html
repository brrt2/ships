001package com.epam.ships.server.gamestates;
002
003import com.epam.ships.shared.fleet.Fleet;
004import com.epam.ships.shared.fleet.Mast;
005import com.epam.ships.shared.fleet.Ship;
006import com.epam.ships.shared.infra.communication.api.Message;
007import com.epam.ships.shared.infra.communication.core.message.MessageBuilder;
008import com.epam.ships.server.CommunicationBus;
009import com.epam.ships.server.WrappedClient;
010import com.epam.ships.server.gamestates.play.PlayState;
011import org.testng.annotations.Test;
012
013import java.util.Arrays;
014
015import static org.mockito.Mockito.mock;
016import static org.mockito.Mockito.when;
017import static org.testng.Assert.*;
018@Test
019public class FleetPlacementStateTest {
020
021  public void shouldProcessState() {
022    //given
023    CommunicationBus communicationBus = mock(CommunicationBus.class);
024    WrappedClient player1 = mock(WrappedClient.class);
025    WrappedClient player2 = mock(WrappedClient.class);
026    when(communicationBus.getFirstClient()).thenReturn(player1);
027    when(communicationBus.getSecondClient()).thenReturn(player2);
028
029    Message fleetMsg = new MessageBuilder()
030        .withFleet(Fleet.ofShips(Arrays.asList(Ship.ofMasts(Mast.ofIndex("3"), Mast.ofIndex("2"), Mast.ofIndex("1")))))
031        .build();
032    when(communicationBus.receive(player1)).thenReturn(fleetMsg);
033    when(communicationBus.receive(player2)).thenReturn(fleetMsg);
034
035    //when
036    GameState playState = new FleetPlacementState(communicationBus).process();
037
038    //then
039    assertNotNull(playState);
040    assertEquals(PlayState.class, playState.getClass());
041  }
042}
