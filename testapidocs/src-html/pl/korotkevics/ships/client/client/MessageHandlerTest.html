001package pl.korotkevics.ships.client.client;
002
003import pl.korotkevics.ships.client.JavaFxInitializer;
004import pl.korotkevics.ships.shared.infra.communication.api.Message;
005import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
006import pl.korotkevics.ships.shared.infra.communication.api.message.Status;
007import pl.korotkevics.ships.shared.infra.communication.core.message.MessageBuilder;
008import javafx.scene.control.Button;
009import org.testng.SkipException;
010import org.testng.annotations.BeforeClass;
011import org.testng.annotations.BeforeMethod;
012import org.testng.annotations.Test;
013
014import java.util.EnumMap;
015import java.util.Map;
016
017import static org.mockito.Mockito.mock;
018import static org.mockito.Mockito.times;
019import static org.mockito.Mockito.verify;
020import static org.testng.Assert.assertTrue;
021
022public class MessageHandlerTest {
023
024  @BeforeClass
025  public void setUp() throws InterruptedException {
026    if(!JavaFxInitializer.isLaunched()) {
027      JavaFxInitializer.initialize();
028    }
029  }
030
031  @BeforeMethod
032  public void skipTestIfThereIsNoGraphicsSupport() {
033    if (!JavaFxInitializer.isEnable()){
034      throw new SkipException("skipping test because of lack of support for graphics");
035    }
036  }
037
038  @Test
039  public void shouldCallConnectionEndTrigger() {
040    //given
041    ConnectionEndTrigger connectionEndTrigger = mock(ConnectionEndTrigger.class);
042    MessageHandler messageHandler = produceMessageHandler(connectionEndTrigger);
043    Message message = connectionEndMessage();
044
045    //when
046    messageHandler.handle(message);
047
048    //then
049    verify(connectionEndTrigger, times(1)).fire(messageHandler.getCurrentEventButton(), message);
050  }
051
052  @Test
053  public void shouldReturnTrueAfterTriggerOpponentWithdrawEven() {
054    //given
055    ConnectionEndTrigger connectionEndTrigger = mock(ConnectionEndTrigger.class);
056    MessageHandler messageHandler = produceMessageHandler(connectionEndTrigger);
057    Message message = connectionEndMessage();
058
059    //when
060    messageHandler.handle(message);
061
062    assertTrue(messageHandler.isEndConnectionTriggered());
063  }
064
065  private Message connectionEndMessage() {
066    return new MessageBuilder()
067        .withHeader(Header.CONNECTION)
068        .withStatus(Status.END)
069        .withStatement("End of a message")
070        .build();
071  }
072
073  private MessageHandler produceMessageHandler(EventTrigger eventTrigger) {
074    Map<Header, EventTrigger> triggers = new EnumMap<>(Header.class);
075    triggers.put(Header.CONNECTION, eventTrigger);
076    MessageHandler messageHandler = new MessageHandler(triggers);
077    messageHandler.setCurrentEventButton(new Button());
078    return messageHandler;
079  }
080}
