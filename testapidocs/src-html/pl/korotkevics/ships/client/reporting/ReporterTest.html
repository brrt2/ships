001package pl.korotkevics.ships.client.reporting;
002
003import org.apache.commons.lang3.StringUtils;
004import org.testng.annotations.Test;
005import org.testng.reporters.Files;
006
007import java.io.File;
008import java.io.FileOutputStream;
009import java.io.IOException;
010import java.net.ServerSocket;
011import java.net.Socket;
012import java.util.Scanner;
013import java.util.concurrent.Executors;
014import java.util.concurrent.Future;
015import java.util.concurrent.TimeUnit;
016
017import static org.testng.Assert.assertEquals;
018
019@Test(groups = "integration")
020public class ReporterTest {
021  
022  private final String REPORTING_TO_FILE_CONFIG = "reportingToFile";
023  
024  private final String REPORTING_TO_SOCKET_CONFIG = "reportingToSocket";
025  
026  private final String REPORTING_TO_LOGGER_CONFIG = "reportingToLogger";
027  
028  private final String REPORTING_TO_WINDOW_CONFIG = "reportingToWindow";
029  
030  private final int PORT = 9919;
031  
032  private final String LOG_FILE = "../target/test-reporting.log";
033  
034  public void shouldRecognizeActiveTargetAsFile() {
035    //given
036    Reporter reporter = new Reporter(REPORTING_TO_FILE_CONFIG);
037    //when - then
038    assertEquals(reporter.getCurrentDestination(), ReportingOption.FILE.toString());
039  }
040  
041  public void shouldRecognizeConfiguredFileName() {
042    //given
043    Reporter reporter = new Reporter(REPORTING_TO_FILE_CONFIG);
044    //when - then
045    assertEquals(reporter.getDestinationFileName(), LOG_FILE);
046  }
047  
048  public void shouldReportToFile() throws IOException {
049    //given
050    Reporter reporter = new Reporter(REPORTING_TO_FILE_CONFIG);
051    //when
052    reporter.report("Just a casual message");
053    //then
054    assertEquals(Files.readFile(new File(LOG_FILE)), "Just a casual message" + System
055                                                                                   .lineSeparator
056                                                                                        ());
057  }
058  
059  public void shouldAppendLinesWhenReportingToFile() throws IOException {
060    //given
061    Reporter reporter = new Reporter(REPORTING_TO_FILE_CONFIG);
062    this.makeSureLogFileExistsAndIsEmpty();
063    //when
064    reporter.report("First message");
065    reporter.report("Second message");
066    //then
067    assertEquals(Files.readFile(new File(LOG_FILE)), "First message" + System.lineSeparator() +
068                                                         "Second message" + System.lineSeparator());
069  }
070  
071  public void shouldNotReportToFile() throws IOException {
072    //given
073    Reporter reporter = new Reporter(REPORTING_TO_WINDOW_CONFIG);
074    this.makeSureLogFileExistsAndIsEmpty();
075    //when
076    reporter.report("A message which should not be reported int" + "o a file since it is not the " +
077                        "" + "" + "" + "" + "" + "" + "" + "active destination.");
078    //then
079    assertEquals(Files.readFile(new File(LOG_FILE)), StringUtils.EMPTY);
080  }
081  
082  public void shouldRecognizeActiveTargetAsSocket() {
083    //given
084    Reporter reporter = new Reporter(REPORTING_TO_SOCKET_CONFIG);
085    //when - then
086    assertEquals(reporter.getCurrentDestination(), ReportingOption.SOCKET.toString());
087  }
088  
089  public void shouldRecognizeHost() {
090    //given
091    Reporter reporter = new Reporter(REPORTING_TO_SOCKET_CONFIG);
092    //when - then
093    assertEquals(reporter.getDestinationHost(), "127.0.0.1");
094  }
095  
096  public void shouldRecognizePort() {
097    //given
098    Reporter reporter = new Reporter(REPORTING_TO_SOCKET_CONFIG);
099    //when - then
100    assertEquals(reporter.getDestinationPort(), PORT);
101  }
102  
103  public void shouldReportToSocket() throws Exception {
104    //given
105    Reporter reporter = new Reporter(REPORTING_TO_SOCKET_CONFIG);
106    Future<String> future = Executors.newSingleThreadExecutor().submit(() -> {
107      try (ServerSocket serverSocket = new ServerSocket(PORT); Socket clientSocket = serverSocket
108                                                                                         .accept
109                                                                                              ()) {
110        Scanner scanner = new Scanner(clientSocket.getInputStream(), "UTF-8");
111        StringBuilder stringBuilder = new StringBuilder();
112        stringBuilder.append(scanner.nextLine());
113        //then
114        return stringBuilder.toString();
115      } catch (final IOException e) {
116        throw new RuntimeException();
117      }
118    });
119    
120    //when
121    reporter.report("Hey Sandor");
122    String receivedString = future.get(5, TimeUnit.SECONDS);// timeout is optional here
123    
124    //then
125    assertEquals(receivedString, "Hey Sandor");
126  }
127  
128  public void shouldRecognizeActiveTargetAsLogger() {
129    //given
130    Reporter reporter = new Reporter(REPORTING_TO_LOGGER_CONFIG);
131    //when - then
132    assertEquals(reporter.getCurrentDestination(), ReportingOption.LOGGER.toString());
133  }
134  
135  public void shouldRecognizeLoggerName() {
136    //given
137    Reporter reporter = new Reporter(REPORTING_TO_LOGGER_CONFIG);
138    //when - then
139    assertEquals(reporter.getDestinationLoggerName(), "Snarky Reporter");
140  }
141  
142  public void shouldRecognizeDestinationWindowTitle() {
143    //given
144    Reporter reporter = new Reporter(REPORTING_TO_WINDOW_CONFIG);
145    //when - then
146    assertEquals(reporter.getDestinationWindowTitle(), "Snarky Reporter");
147  }
148  
149  public void shouldRecognizeDestinationWindowHeight() {
150    //given
151    Reporter reporter = new Reporter(REPORTING_TO_WINDOW_CONFIG);
152    //when - then
153    assertEquals(reporter.getDestinationWindowHeight(), 600);
154  }
155  
156  public void shouldRecognizeDestinationWindowWidth() {
157    //given
158    Reporter reporter = new Reporter(REPORTING_TO_WINDOW_CONFIG);
159    //when - then
160    assertEquals(reporter.getDestinationWindowWidth(), 400);
161  }
162  
163  private void makeSureLogFileExistsAndIsEmpty() throws IOException {
164    FileOutputStream writer = new FileOutputStream(LOG_FILE);
165    writer.write(("").getBytes());
166    writer.close();
167  }
168}
