001package pl.korotkevics.ships.shared.infra.communication.core.message;
002
003import org.testng.annotations.Test;
004import pl.korotkevics.ships.shared.fleet.Fleet;
005import pl.korotkevics.ships.shared.fleet.Mast;
006import pl.korotkevics.ships.shared.fleet.Ship;
007import pl.korotkevics.ships.shared.infra.communication.api.Message;
008import pl.korotkevics.ships.shared.infra.communication.api.message.Author;
009import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
010import pl.korotkevics.ships.shared.infra.communication.api.message.Status;
011
012import java.util.Arrays;
013import java.util.List;
014
015import static org.testng.Assert.assertEquals;
016import static org.testng.Assert.assertFalse;
017import static org.testng.Assert.assertNotEquals;
018import static org.testng.Assert.assertTrue;
019
020@Test
021public class BaseMessageTest {
022  
023  public void stringRepresentationIsProper() {
024    //given - when
025    Message message = new MessageBuilder().withAuthor(Author.SERVER).withStatement("Hello World")
026                          .withHeader(Header.WIN).build();
027    //then
028    assertEquals(message.toString(), "BaseMessage(header=WIN, status=OK, author=SERVER, " +
029                                         "statement=Hello World, fleet=Fleet(fleet={}))");
030  }
031  
032  public void defaultMessagesAreEqual() {
033    //given
034    Message firstMessageToCompare = new MessageBuilder().build();
035    Message secondMessageToCompare = new MessageBuilder().build();
036    //when - then
037    assertEquals(firstMessageToCompare, secondMessageToCompare);
038  }
039  
040  
041  public void defaultMessagesHaveSameHashcode() {
042    //given
043    Message firstMessageToCompare = new MessageBuilder().build();
044    Message secondMessageToCompare = new MessageBuilder().build();
045    //when - then
046    assertTrue(firstMessageToCompare.hashCode()==secondMessageToCompare.hashCode());
047  }
048  
049  public void equalsWorksAsExpectedForSameMessages() {
050    //given - when
051    Message firstMessageToCompare = new MessageBuilder()
052                                        .withFleet(this.produceFleet())
053                                        .withAuthor(Author.CLIENT)
054                                        .withHeader(Header.MANUAL_PLACEMENT)
055                                        .withStatement("Hey hey hey")
056                                        .build();
057  
058    Message secondMessageToCompare = new MessageBuilder()
059                                        .withFleet(this.produceFleet())
060                                        .withAuthor(Author.CLIENT)
061                                        .withHeader(Header.MANUAL_PLACEMENT)
062                                        .withStatement("Hey hey hey")
063                                        .build();
064    //then
065    assertEquals(firstMessageToCompare, secondMessageToCompare);
066  }
067  
068  public void equalsWorksAsExpectedForDifferentMessages() {
069    //given - when
070    Message firstMessageToCompare = new MessageBuilder()
071                                        .withFleet(this.produceFleet())
072                                        .withHeader(Header.RANDOM_PLACEMENT)
073                                        .build();
074    
075    Message secondMessageToCompare = new MessageBuilder()
076                                         .withFleet(this.produceFleet())
077                                         .withHeader(Header.MANUAL_PLACEMENT)
078                                         .withStatement("Yeah")
079                                         .build();
080    //then
081    assertNotEquals(firstMessageToCompare, secondMessageToCompare);
082  }
083  
084  public void hashCodesAreEqualForSameMessages() {
085    //given - when
086    Message firstMessageToCompare = new MessageBuilder()
087                                        .withFleet(this.produceFleet())
088                                        .withAuthor(Author.CLIENT)
089                                        .withHeader(Header.MANUAL_PLACEMENT)
090                                        .withStatement("CAFEBABE")
091                                        .build();
092  
093    Message secondMessageToCompare = new MessageBuilder()
094                                         .withFleet(this.produceFleet())
095                                         .withAuthor(Author.CLIENT)
096                                         .withHeader(Header.MANUAL_PLACEMENT)
097                                         .withStatement("CAFEBABE")
098                                         .build();
099    //then
100    assertTrue(firstMessageToCompare.hashCode() == secondMessageToCompare.hashCode());
101  }
102  
103  public void hashCodesAreNotEqualForDifferentMessages() {
104    //given - when
105    Message firstMessageToCompare = new MessageBuilder()
106                                        .withFleet(this.produceFleet())
107                                        .withAuthor(Author.SERVER)
108                                        .withHeader(Header.MANUAL_PLACEMENT)
109                                        .withStatement("SCALA")
110                                        .build();
111    
112    Message secondMessageToCompare = new MessageBuilder()
113                                         .withFleet(this.produceFleet())
114                                         .withHeader(Header.MANUAL_PLACEMENT)
115                                         .withStatement("OPENJDK")
116                                         .build();
117    //then
118    assertFalse(firstMessageToCompare.hashCode() == secondMessageToCompare.hashCode());
119  }
120  
121  
122  private Fleet produceFleet() {
123    List<Ship> ships = Arrays.asList(Ship.ofMasts(Mast.ofIndex("5")),
124        Ship.ofMasts(Mast.ofIndex("2"), Mast.ofIndex("3"), Mast.ofIndex("1")),
125        Ship.ofMasts(Mast.ofIndex("21"), Mast.ofIndex("23"), Mast.ofIndex("24")));
126    return Fleet.ofShips(ships);
127  }
128  
129}
