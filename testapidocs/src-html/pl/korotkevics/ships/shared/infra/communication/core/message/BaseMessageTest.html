001package pl.korotkevics.ships.shared.infra.communication.core.message;
002
003import org.testng.annotations.Test;
004import pl.korotkevics.ships.shared.fleet.Fleet;
005import pl.korotkevics.ships.shared.fleet.Mast;
006import pl.korotkevics.ships.shared.fleet.Ship;
007import pl.korotkevics.ships.shared.infra.communication.api.Message;
008import pl.korotkevics.ships.shared.infra.communication.api.message.Author;
009import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
010import pl.korotkevics.ships.shared.infra.communication.api.message.Status;
011
012import java.util.Arrays;
013import java.util.List;
014
015import static org.testng.Assert.assertEquals;
016import static org.testng.Assert.assertTrue;
017
018@Test
019public class BaseMessageTest {
020  
021  public void stringRepresentationIsProper() {
022    //given - when
023    Message message = new MessageBuilder().withAuthor(Author.SERVER).withStatement("Hello World")
024                          .withHeader(Header.WIN).build();
025    //then
026    assertEquals(message.toString(), "BaseMessage(header=WIN, status=OK, author=SERVER, " +
027                                         "statement=Hello World, fleet=Fleet(fleet={}))");
028  }
029  
030  public void equalsWorksAsExpected() {
031    //given - when
032    Message firstMessageToCompare = new MessageBuilder()
033                                        .withFleet(this.produceFleet())
034                                        .withAuthor(Author.CLIENT)
035                                        .withHeader(Header.MANUAL_PLACEMENT)
036                                        .withStatement("Hey hey hey")
037                                        .build();
038  
039    Message secondMessageToCompare = new MessageBuilder()
040                                        .withFleet(this.produceFleet())
041                                        .withAuthor(Author.CLIENT)
042                                        .withHeader(Header.MANUAL_PLACEMENT)
043                                        .withStatement("Hey hey hey")
044                                        .build();
045    //then
046    assertEquals(firstMessageToCompare, secondMessageToCompare);
047  }
048  
049  public void hashCodeWorksAsExpected() {
050    //given - when
051    Message firstMessageToCompare = new MessageBuilder()
052                                        .withFleet(this.produceFleet())
053                                        .withAuthor(Author.CLIENT)
054                                        .withHeader(Header.MANUAL_PLACEMENT)
055                                        .withStatement("CAFEBABE")
056                                        .build();
057  
058    Message secondMessageToCompare = new MessageBuilder()
059                                         .withFleet(this.produceFleet())
060                                         .withAuthor(Author.CLIENT)
061                                         .withHeader(Header.MANUAL_PLACEMENT)
062                                         .withStatement("CAFEBABE")
063                                         .build();
064    //then
065    assertTrue(firstMessageToCompare.hashCode() == secondMessageToCompare.hashCode());
066  }
067  
068  private Fleet produceFleet() {
069    List<Ship> ships = Arrays.asList(Ship.ofMasts(Mast.ofIndex("5")),
070        Ship.ofMasts(Mast.ofIndex("2"), Mast.ofIndex("3"), Mast.ofIndex("1")),
071        Ship.ofMasts(Mast.ofIndex("21"), Mast.ofIndex("23"), Mast.ofIndex("24")));
072    return Fleet.ofShips(ships);
073  }
074  
075}
