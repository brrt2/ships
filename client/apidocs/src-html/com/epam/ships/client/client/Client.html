001package com.epam.ships.client.client;
002
003import com.epam.ships.fleet.Fleet;
004import com.epam.ships.infra.communication.api.Message;
005import com.epam.ships.infra.communication.api.io.Receiver;
006import com.epam.ships.infra.communication.api.io.Sender;
007import com.epam.ships.infra.communication.api.message.Author;
008import com.epam.ships.infra.communication.api.message.Header;
009import com.epam.ships.infra.communication.core.json.io.JsonReceiver;
010import com.epam.ships.infra.communication.core.json.io.JsonSender;
011import com.epam.ships.infra.communication.core.message.MessageBuilder;
012import com.epam.ships.infra.logging.api.Target;
013import com.epam.ships.infra.logging.core.SharedLogger;
014import javafx.scene.control.Button;
015
016import java.io.IOException;
017import java.net.InetAddress;
018import java.net.InetSocketAddress;
019import java.net.Socket;
020import java.util.EnumMap;
021import java.util.Map;
022
023/**
024 * Enable communication with server side.
025 *
026 * @author Magdalena Aarsman
027 * @since 2017-12-13
028 */
029public class Client implements Runnable {
030  private static final Target logger = new SharedLogger(Client.class);
031  private final MessageHandler messageHandler;
032  private Socket clientSocket;
033  private volatile boolean shouldRun;
034
035  /**
036   * Creates Client instance.
037   * @param messageHandler - messageHandler instance
038   * @param shouldRun - flag informing if client should start/run
039   */
040  public Client(final MessageHandler messageHandler, boolean shouldRun) {
041    this.messageHandler = messageHandler;
042    this.shouldRun = shouldRun;
043  }
044
045  /**
046   * Connecting client with server using providing address and port.
047   *
048   * @param ipAddress - server ip address
049   * @param port      - server port
050   * @param socket    - socket to connect on.
051   * @return true if success, false on failure
052   */
053  public boolean connect(final String ipAddress, final int port, Socket socket) {
054    try {
055      clientSocket = socket;
056      final InetAddress address = InetAddress.getByName(ipAddress);
057      final int connectionTimeout = 500;
058      clientSocket.connect(new InetSocketAddress(address, port), connectionTimeout);
059    } catch (IOException | IllegalArgumentException e) {
060      logger.error(e.getMessage());
061      return false;
062    }
063    return true;
064  }
065
066  /**
067   * Override run method of Runnable interface.
068   * Setting name of thread to "Client listen Thread" and start listen loop.
069   */
070  @Override
071  public void run() {
072    Thread.currentThread().setName("Client listen Thread");
073    listenLoop();
074  }
075
076  /**
077   * Closing client, stopping listen thread.
078   */
079  public void closeClient() {
080    this.shouldRun = false;
081    if (clientSocket == null) {
082      return;
083    }
084
085    try {
086      clientSocket.close();
087    } catch (IOException e) {
088      logger.error(e.getMessage());
089    }
090  }
091
092  private void listenLoop() {
093
094    while (shouldRun && !messageHandler.isEndConnectionTriggered()) {
095      try {
096        Receiver receiver = new JsonReceiver(clientSocket.getInputStream());
097        Message message = receiver.receive();
098
099        logger.info(message);
100
101        messageHandler.handle(message);
102
103      } catch (IOException | IllegalStateException e) {
104        logger.error(e.getMessage());
105      }
106
107      rest();
108    }
109  }
110
111  private void rest() {
112    try {
113      Thread.sleep(300);
114    } catch (InterruptedException e) {
115      logger.info(e.getMessage());
116      Thread.currentThread().interrupt();
117    }
118  }
119
120  /**
121   * Sending message containing shot index to server.
122   *
123   * @param shotIndex - index of last shot
124   */
125  public void sendShot(int shotIndex) {
126    try {
127      Sender sender = new JsonSender(clientSocket.getOutputStream());
128      Message shot = new MessageBuilder().withHeader(Header.SHOT)
129          .withAuthor(Author.CLIENT)
130          .withStatement(String.valueOf(shotIndex))
131          .build();
132      sender.send(shot);
133    } catch (IOException e) {
134      logger.error(e.getMessage());
135    }
136  }
137
138  /**
139   * Sending message containing fleet placement to server.
140   *
141   * @param fleet - Fleet object representing user fleet.
142   */
143  public void sendFleet(Fleet fleet) {
144    try {
145      Sender sender = new JsonSender(clientSocket.getOutputStream());
146      Message fleetMsg = new MessageBuilder()
147          .withHeader(Header.PLACEMENT)
148          .withAuthor(Author.CLIENT)
149          .withFleet(fleet)
150          .build();
151      sender.send(fleetMsg);
152    } catch (IOException e) {
153      logger.error(e.getMessage());
154    }
155  }
156
157  /**
158   * Set handle to gui to trigger some events in response for server message.
159   *
160   * @param button - gui handler as button.
161   */
162  public void setEventTrigger(Button button) {
163    messageHandler.setCurrentEventButton(button);
164  }
165}
