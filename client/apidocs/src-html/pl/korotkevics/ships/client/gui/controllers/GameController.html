001package pl.korotkevics.ships.client.gui.controllers;
002
003import pl.korotkevics.ships.client.client.Client;
004import pl.korotkevics.ships.client.gui.events.HitShotEvent;
005import pl.korotkevics.ships.client.gui.events.LooseEvent;
006import pl.korotkevics.ships.client.gui.events.MissShotEvent;
007import pl.korotkevics.ships.client.gui.events.OpponentShotEvent;

009import pl.korotkevics.ships.client.gui.events.TurnChangeEvent;
010import pl.korotkevics.ships.client.gui.events.WinEvent;
011import pl.korotkevics.ships.shared.infra.logging.api.Target;
012import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;
013import javafx.beans.binding.Bindings;

017import javafx.geometry.HPos;

019import javafx.scene.control.Button;

021import javafx.scene.input.MouseEvent;

023import javafx.scene.layout.GridPane;

028import java.io.IOException;
029
030/**
031 * Game window controller.
032 * @author Magdalena Aarsman
033 * @since 2017-12-15
034 */
035
036public class GameController {
037
038  private static final Target logger = new SharedLogger(Client.class);
039
040  private static final int BOARD_SIZE = 10;
041
042  @FXML
043  private GridPane yourBoard;
044
045  @FXML
046  private GridPane opponentBoard;
047
048  @FXML
049  private AnchorPane mainAnchorPane;
050
051  @FXML
052  private Button eventButton;
053
054  @FXML
055  private Label winLabel;
056
057  private int shotIndex;
058
059  @FXML
060  void initialize() {
061    eventButton.addEventHandler(OpponentWithdrawEvent.OPPONENT_WITHDRAW,
062        opponentConnectedEvent -> opponentWithdraw());
063    eventButton.addEventHandler(OpponentShotEvent.OPPONENT_SHOT,
064        opponentShotEvent -> setOpponentShot(opponentShotEvent.getShotIndex()));
065    eventButton.addEventHandler(TurnChangeEvent.TURN_EVENT, event -> setMyTurn());
066    eventButton.addEventHandler(MissShotEvent.MISS_SHOT, event -> changeTurn());
067    eventButton.addEventHandler(HitShotEvent.HIT_SHOT, event -> setHit());
068    eventButton.addEventHandler(WinEvent.GAME_WIN, event -> setWin());
069    eventButton.addEventHandler(LooseEvent.GAME_LOSE, event -> setLose());
070
071    initializeTurn(false);
072  }
073
074  private void initializeTurn(boolean myTurn) {
075    logger.info("initialize turn: " + myTurn);
076
077    if (!myTurn) {
078      opponentBoard.setDisable(true);
079      final double opacity = 0.4;
080      opponentBoard.setOpacity(opacity);
081    }
082  }
083
084  void initializeClient() {
085    getClient().setEventTrigger(eventButton);
086  }
087
088  void initializeBoards(GridPane yourBoard) {
089    final int margin = 50;
090    final NumberBinding allRectanglesWidth = Bindings.min(this.yourBoard.heightProperty(),
091        this.yourBoard.widthProperty().add(-margin));
092    final NumberBinding allRectanglesHeight = Bindings.min(this.yourBoard.heightProperty(),
093        this.yourBoard.widthProperty()).add(-margin);
094
095    for (int i = 0; i < BOARD_SIZE; i++) {
096      for (int j = 0; j < BOARD_SIZE; j++) {
097        final int shotIndex = j * BOARD_SIZE + i;
098
099        Rectangle yourRect = getYourRect(allRectanglesWidth, allRectanglesHeight);
100        Rectangle opponentRect = getOpponentRectangle(allRectanglesWidth,
101            allRectanglesHeight, shotIndex);
102
103        this.yourBoard.add(yourRect, i, j);
104        opponentBoard.add(opponentRect, i, j);
105
106        GridPane.setHalignment(yourRect, HPos.CENTER);
107        GridPane.setHalignment(opponentRect, HPos.CENTER);
108      }
109    }
110
111    copyYourBoard(yourBoard);
112  }
113
114  private void copyYourBoard(GridPane yourBoard) {
115    for (int i = 1; i < yourBoard.getChildren().size(); i++) {
116      if (((Rectangle) yourBoard.getChildren().get(i)).getFill() == Color.GREEN) {
117        ((Rectangle) this.yourBoard.getChildren().get(i)).setFill(Color.GREEN);
118      }
119    }
120  }
121
122  private Client getClient() {
123    MainController mainController = (MainController) mainAnchorPane.getParent().getUserData();
124    return mainController.getClient();
125  }
126
127  private Rectangle getYourRect(NumberBinding allRectanglesWidth,
128                                NumberBinding allRectanglesHeight) {
129    final int initialSize = 15;
130    Rectangle yourRect = new Rectangle(initialSize, initialSize, Color.GRAY);
131    yourRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
132    yourRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
133
134    return yourRect;
135  }
136
137  private Rectangle getOpponentRectangle(NumberBinding allRectanglesWidth,
138                                         NumberBinding allRectanglesHeight,
139                                         final int shotIndex) {
140    final int initialSize = 15;
141    Rectangle opponentRect = new Rectangle(initialSize, initialSize, Color.GRAY);
142
143    opponentRect.setOnMouseClicked((MouseEvent mouseEvent) -> {
144      opponentRect.setFill(Color.BLACK);
145      logger.info(shotIndex);
146      this.shotIndex = shotIndex;
147      getClient().sendShot(shotIndex);
148    });
149
150    opponentRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
151    opponentRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
152
153    return opponentRect;
154  }
155
156  private void opponentWithdraw() {
157    getClient().closeClient();
158    loadWithdrawScreen();
159  }
160
161  private void loadWithdrawScreen() {
162    try {
163      final String opponentWithdrawUrl = "/fxml/opponentWithdraw.fxml";
164      final FXMLLoader opponentWithdrawLoader =
165          new FXMLLoader(getClass().getResource(opponentWithdrawUrl));
166      final Parent opponentWithdraw = opponentWithdrawLoader.load();
167      final AnchorPane mainPane = (AnchorPane) mainAnchorPane.getParent();
168      final int sceneHeight = 400;
169      final int sceneWidth = 600;
170      final double margin = 0.0;
171
172      Stage stage = (Stage) mainPane.getScene().getWindow();
173      stage.setHeight(sceneHeight);
174      stage.setWidth(sceneWidth);
175
176      mainPane.getChildren().clear();
177      mainPane.getChildren().setAll(opponentWithdraw);
178
179      AnchorPane.setTopAnchor(opponentWithdraw, margin);
180      AnchorPane.setBottomAnchor(opponentWithdraw, margin);
181      AnchorPane.setLeftAnchor(opponentWithdraw, margin);
182      AnchorPane.setRightAnchor(opponentWithdraw, margin);
183
184    } catch (IOException e) {
185      logger.error(e.getMessage());
186    }
187  }
188
189  private void setOpponentShot(int shotIndex) {
190    int column = shotIndex / BOARD_SIZE;
191    int row = shotIndex - (column * BOARD_SIZE);
192    int newShotIndex = row * BOARD_SIZE + column;
193    logger.info("new shotIndex: " + newShotIndex);
194    Rectangle rec = (Rectangle) (yourBoard.getChildren().get(newShotIndex + 1));
195    if (rec.getFill() == Color.GREEN) {
196      rec.setFill(Color.RED);
197    } else {
198      rec.setFill(Color.BLACK);
199    }
200  }
201
202  private void changeTurn() {
203    final double opacity = 0.4;
204    opponentBoard.setDisable(true);
205    opponentBoard.setOpacity(opacity);
206  }
207
208  private void setHit() {
209    int column = shotIndex / BOARD_SIZE;
210    int row = shotIndex - (column * BOARD_SIZE);
211    int newShotIndex = row * BOARD_SIZE + column;
212    Rectangle rec = (Rectangle) (opponentBoard.getChildren().get(newShotIndex + 1));
213    rec.setFill(Color.RED);
214  }
215
216  private void setMyTurn() {
217    logger.info("set my turn game controller");
218    final double noOpacity = 1.0;
219    opponentBoard.setDisable(false);
220    opponentBoard.setOpacity(noOpacity);
221  }
222
223  private void setWin() {
224    getClient().closeClient();
225    eventButton.removeEventHandler(OpponentWithdrawEvent.OPPONENT_WITHDRAW,
226        opponentConnectedEvent -> opponentWithdraw());
227    winLabel.setStyle("-fx-color: green");
228    winLabel.setText("YOU WIN!");
229    final double opacity = 0.4;
230    yourBoard.setDisable(true);
231    yourBoard.setOpacity(opacity);
232    opponentBoard.setDisable(true);
233    opponentBoard.setOpacity(opacity);
234  }
235
236  private void setLose() {
237    getClient().closeClient();
238    eventButton.removeEventHandler(OpponentWithdrawEvent.OPPONENT_WITHDRAW,
239        opponentConnectedEvent -> opponentWithdraw());
240    winLabel.setStyle("-fx-color: green");
241    winLabel.setText("YOU LOSE!");
242    final double opacity = 0.4;
243    yourBoard.setDisable(true);
244    yourBoard.setOpacity(opacity);
245    opponentBoard.setDisable(true);
246    opponentBoard.setOpacity(opacity);
247  }
248}
