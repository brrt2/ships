001package pl.korotkevics.ships.client.gui.events;
002
003import javafx.event.Event;
004import javafx.event.EventType;
005import lombok.Getter;
006import pl.korotkevics.ships.shared.fleet.Fleet;
007
008/**
009 * Event firing when a fleet is placed randomly.
010 * @author Magdalena Aarsman
011 * @since 2017-01-15
012 */
013public class RandomPlacementEvent extends Event {
014  public static final EventType<RandomPlacementEvent> RANDOM_PLACEMENT_EVENT =
015      new EventType<>(javafx.event.Event.ANY, "RANDOM_PLACEMENT_EVENT");
016
017  @Getter
018  private final Fleet fleet;
019
020  public RandomPlacementEvent(final Fleet fleet) {
021    super(RANDOM_PLACEMENT_EVENT);
022    this.fleet = fleet;
023  }
024}
