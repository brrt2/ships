001package pl.korotkevics.ships.client.gui.controllers;
002
003import javafx.event.ActionEvent;
004import javafx.fxml.FXML;
005import javafx.fxml.FXMLLoader;
006import javafx.scene.Parent;
007import javafx.scene.Scene;
008import javafx.scene.control.Button;
009import javafx.scene.control.Label;
010import javafx.scene.layout.Pane;
011import javafx.stage.Stage;
012import lombok.Getter;
013import pl.korotkevics.ships.client.client.Client;
014import pl.korotkevics.ships.client.gui.util.UTF8Control;
015import pl.korotkevics.ships.client.localization.OurLocale;
016import pl.korotkevics.ships.shared.infra.logging.api.Target;
017import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;
018
019import java.io.IOException;
020import java.util.Locale;
021import java.util.ResourceBundle;
022
023
024/**
025 * Main window (containing title and similar) controller.
026 *
027 * @author Magdalena Aarsman, Sandor Korotkevics
028 * @since 2017-12-14
029 */
030
031public class MainController {
032
033  private static final Target logger = new SharedLogger(Client.class);
034
035  private static final String DICTIONARY = "dict";
036
037  @FXML
038  private Label gameTitle;
039
040  @FXML
041  private Button russianButton;
042
043  @FXML
044  private Pane mainPane;
045
046  @Getter
047  private Client client;
048
049  @FXML
050  private Button polishButton;
051
052  @FXML
053  private Button englishButton;
054
055  private ResourceBundle resourceBundle;
056
057  /**
058   * Initialize main controller.
059   *
060   * @param client - Instance of Client, which communicate with server.
061   */
062  @FXML
063  public void initialize(final Client client) {
064    this.client = client;
065    this.loadDefaultView();
066
067  }
068
069  private void loadDefaultView() {
070    this.reloadView();
071  }
072
073  private void reloadView() {
074    final FXMLLoader fxmlLoader = this.prepareFxmlLoader();
075    this.gameTitle.setText(this.resourceBundle.getString("gameTitle"));
076    try {
077      this.addToMainPane(fxmlLoader.load());
078    } catch (IOException e) {
079      logger.error(e.getMessage());
080    }
081  }
082
083  private FXMLLoader prepareFxmlLoader() {
084    final String url = "/fxml/connectWindow.fxml";
085    final FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource(url));
086    this.resourceBundle = ResourceBundle.getBundle(DICTIONARY, new UTF8Control());
087    fxmlLoader.setResources(this.resourceBundle);
088    return fxmlLoader;
089  }
090
091  private void addToMainPane(final Parent parent) {
092    this.mainPane.getChildren().clear();
093    this.mainPane.getChildren().add(parent);
094  }
095
096  @FXML
097  void triggerPolishVersion(final ActionEvent event) {
098    logger.info("The language is set to " + OurLocale.POLISH);
099    Locale.setDefault(new Locale(OurLocale.POLISH.toString()));
100    this.reloadView();
101    this.changeTitleLabel();
102  }
103
104  private void changeTitleLabel() {
105    Scene scene = mainPane.getScene();
106    Stage stage = (Stage) scene.getWindow();
107    stage.setTitle(this.resourceBundle.getString("gameTitle"));
108  }
109
110  @FXML
111  void triggerEnglishVersion(final ActionEvent event) {
112    logger.info("The language is set to " + OurLocale.ENGLISH);
113    Locale.setDefault(new Locale(OurLocale.ENGLISH.toString()));
114    this.reloadView();
115    this.changeTitleLabel();
116  }
117
118  @FXML
119  void triggerRussianVersion(final ActionEvent event) {
120    logger.info("The language is set to " + OurLocale.RUSSIAN);
121    Locale.setDefault(new Locale(OurLocale.RUSSIAN.toString()));
122    this.reloadView();
123    this.changeTitleLabel();
124  }
125
126  void disableLocalizationButtons() {
127    this.hideButton(this.polishButton);
128    this.hideButton(this.englishButton);
129    this.hideButton(this.russianButton);
130  }
131
132  private void hideButton(Button button) {
133    button.setDisable(true);
134    button.setVisible(false);
135  }
136}
