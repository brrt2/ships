001package pl.korotkevics.ships.client.gui.util;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.InputStreamReader;
006import java.net.URL;
007import java.net.URLConnection;
008import java.util.Locale;
009import java.util.PropertyResourceBundle;
010import java.util.ResourceBundle;
011import java.util.ResourceBundle.Control;
012
013/**
014 * Used when instantiating ResourceBundle in order
015 * to enforce UTF-8 in UI (to make Polish, Russian
016 * special characters appear properly).
017 * <p>
018 * <p> We're thankful to the author of this code.
019 * The author copied the default implementation and
020 * added "UTF-8" when instantiating InputStreamReader. </p>
021 *
022 * @author https://stackoverflow.com/questions/4659929/how-to-use-utf-8-in-resource-properties
023 * -with-resourcebundle
024 * @since 2011-01-11
025 */
026
027public class UTF8Control extends Control {
028  
029  private static final String ENCODING = "UTF-8";
030  
031  @Override
032  public ResourceBundle newBundle(final String baseName, final Locale locale, final String
033                                                                                    format, final
034  ClassLoader loader, final boolean reload) throws IllegalAccessException, InstantiationException, IOException {
035    // The below is a copy of the default implementation.
036    String bundleName = toBundleName(baseName, locale);
037    String resourceName = toResourceName(bundleName, "properties");
038    ResourceBundle bundle = null;
039    InputStream stream = null;
040    if (reload) {
041      URL url = loader.getResource(resourceName);
042      if (url != null) {
043        URLConnection connection = url.openConnection();
044        if (connection != null) {
045          connection.setUseCaches(false);
046          stream = connection.getInputStream();
047        }
048      }
049    } else {
050      stream = loader.getResourceAsStream(resourceName);
051    }
052    if (stream != null) {
053      try {
054        // Only this line is changed to make it to read properties files as UTF-8.
055        bundle = new PropertyResourceBundle(new InputStreamReader(stream, ENCODING));
056      } finally {
057        stream.close();
058      }
059    }
060    return bundle;
061  }
062}
