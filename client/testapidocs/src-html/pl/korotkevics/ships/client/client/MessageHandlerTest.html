001package pl.korotkevics.ships.client.client;
002
003import javafx.scene.control.Button;
004import org.testng.SkipException;
005import org.testng.annotations.BeforeClass;
006import org.testng.annotations.BeforeMethod;
007import org.testng.annotations.Test;
008import pl.korotkevics.ships.client.JavaFxInitializer;
009import pl.korotkevics.ships.client.reporting.Reporter;
010import pl.korotkevics.ships.shared.infra.communication.api.Message;
011import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
012import pl.korotkevics.ships.shared.infra.communication.api.message.Status;
013import pl.korotkevics.ships.shared.infra.communication.core.message.MessageBuilder;
014
015import java.util.EnumMap;
016import java.util.Map;
017
018import static org.mockito.Mockito.mock;
019import static org.mockito.Mockito.times;
020import static org.mockito.Mockito.verify;
021import static org.testng.Assert.assertTrue;
022
023public class MessageHandlerTest {
024  
025  @BeforeClass
026  public void setUp() throws InterruptedException {
027    if (!JavaFxInitializer.isLaunched()) {
028      JavaFxInitializer.initialize();
029    }
030  }
031  
032  @BeforeMethod
033  public void skipTestIfThereIsNoGraphicsSupport() {
034    if (!JavaFxInitializer.isEnable()) {
035      throw new SkipException("skipping test because of lack of support for graphics");
036    }
037  }
038  
039  @Test
040  public void shouldCallConnectionEndTrigger() {
041    //given
042    ConnectionEndTrigger connectionEndTrigger = mock(ConnectionEndTrigger.class);
043    MessageHandler messageHandler = produceMessageHandler(connectionEndTrigger);
044    Message message = connectionEndMessage();
045    
046    //when
047    messageHandler.handle(message);
048    
049    //then
050    verify(connectionEndTrigger, times(1)).fire(messageHandler.getCurrentEventButton(), message);
051  }
052  
053  @Test
054  public void shouldReturnTrueAfterTriggerOpponentWithdrawEven() {
055    //given
056    ConnectionEndTrigger connectionEndTrigger = mock(ConnectionEndTrigger.class);
057    MessageHandler messageHandler = produceMessageHandler(connectionEndTrigger);
058    Message message = connectionEndMessage();
059    
060    //when
061    messageHandler.handle(message);
062    
063    assertTrue(messageHandler.isEndConnectionTriggered());
064  }
065  
066  private Message connectionEndMessage() {
067    return new MessageBuilder().withHeader(Header.CONNECTION).withStatus(Status.END)
068               .withStatement("End of a message").build();
069  }
070  
071  private MessageHandler produceMessageHandler(EventTrigger eventTrigger) {
072    Map<Header, EventTrigger> triggers = new EnumMap<>(Header.class);
073    Reporter reporter = new Reporter("");
074    triggers.put(Header.CONNECTION, eventTrigger);
075    MessageHandler messageHandler = new MessageHandler(triggers, reporter);
076    messageHandler.setCurrentEventButton(new Button());
077    return messageHandler;
078  }
079}
