001package pl.korotkevics.ships.client.validators;
002
003import pl.korotkevics.ships.client.gui.util.FieldState;
004import pl.korotkevics.ships.client.gui.util.ShipOrientation;
005import org.testng.annotations.DataProvider;
006import org.testng.annotations.Test;
007
008import java.util.Arrays;
009import java.util.List;
010
011import static org.testng.Assert.assertFalse;
012import static org.testng.Assert.assertTrue;
013
014public class ShipPlacementValidatorTest {
015
016  private static final int BOARD_SIZE = 10;
017
018  @DataProvider(name = "validPlacement")
019  public static Object [][] validData() {
020    return new Object[][] {
021        {ShipOrientation.VERTICAL, 0, 3, producedBoard(Arrays.asList(99,81))},
022        {ShipOrientation.HORIZONTAL, 0, 3, producedBoard(Arrays.asList(4,2,3))},
023        {ShipOrientation.VERTICAL, 0, 4, producedBoard(Arrays.asList(20,30,40))}
024    };
025  }
026
027  @DataProvider(name = "invalidPlacement")
028  public static Object [][] invalidData() {
029    return new Object[][] {
030        {ShipOrientation.HORIZONTAL, 0, 3, producedBoard(Arrays.asList(0,1,2,3))},
031        {ShipOrientation.VERTICAL, 0, 3, producedBoard(Arrays.asList(0,1,2,3))},
032        {ShipOrientation.HORIZONTAL, 25, 2, producedBoard(Arrays.asList(45,1,2,3))},
033        {ShipOrientation.VERTICAL, 25, 2, producedBoard(Arrays.asList(15,1,2,3))},
034        {ShipOrientation.HORIZONTAL, 25, 2, producedBoard(Arrays.asList(24,1,2,3))},
035        {ShipOrientation.VERTICAL, 25, 2, producedBoard(Arrays.asList(27,1,2,3))}
036    };
037  }
038
039
040  @Test(dataProvider = "validPlacement")
041  public void validPlacement(ShipOrientation shipOrientation, int shipStartIndex, int mastCount,
042                             FieldState[] board) {
043    //given
044    ShipPlacementValidator shipPlacementValidator =
045        new ShipPlacementValidator(shipOrientation, shipStartIndex, mastCount, board);
046    //when
047    boolean isValid = shipPlacementValidator.isPlacementValid();
048    //then
049    assertTrue(isValid);
050
051  }
052
053  @Test(dataProvider = "invalidPlacement")
054  void invalidPlacement(ShipOrientation shipOrientation, int shipStartIndex, int mastCount,
055                        FieldState[] board) {
056    //given
057    ShipPlacementValidator shipPlacementValidator =
058        new ShipPlacementValidator(shipOrientation, shipStartIndex, mastCount, board);
059    //when
060    boolean isValid = shipPlacementValidator.isPlacementValid();
061    //then
062    assertFalse(isValid);
063  }
064
065  private static FieldState [] producedBoard(List<Integer> occupiedPlaces) {
066    FieldState [] board = new FieldState[BOARD_SIZE * BOARD_SIZE];
067    for (int i = 0; i < BOARD_SIZE * BOARD_SIZE; i++) {
068      if(occupiedPlaces.contains(Integer.valueOf(i))) {
069        board[i] = FieldState.OCCUPIED;
070      } else {
071        board[i] = FieldState.EMPTY;
072      }
073    }
074
075    return board;
076  }
077}
