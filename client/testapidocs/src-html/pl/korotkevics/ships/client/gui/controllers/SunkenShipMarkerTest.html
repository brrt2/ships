001package pl.korotkevics.ships.client.gui.controllers;
002
003import org.testng.annotations.DataProvider;
004import org.testng.annotations.Test;
005import pl.korotkevics.ships.client.gui.util.FieldState;
006
007import java.util.Arrays;
008import java.util.HashSet;
009import java.util.List;
010import java.util.Set;
011
012import static org.testng.Assert.assertTrue;
013
014public class SunkenShipMarkerTest {
015  private static final int BOARD_SIZE = 10;
016
017  @DataProvider
018  static Object[][] data() {
019    return new Object[][] {
020        {3, produceBoard(Arrays.asList(3,4,5)), new HashSet<Integer>(Arrays.asList(2,6,12,13,14,15,16))},
021        {3, produceBoard(Arrays.asList(3,13,23)), new HashSet<Integer>(Arrays.asList(2,12,22,32,33,34,4,14,24))}
022    };
023  }
024
025  private static FieldState[] produceBoard(final List<Integer> occupiedPlaces) {
026    FieldState[] board = new FieldState[BOARD_SIZE * BOARD_SIZE];
027    for (int i = 0; i < BOARD_SIZE * BOARD_SIZE; i++) {
028      if (occupiedPlaces.contains(Integer.valueOf(i))) {
029        board[i] = FieldState.HIT;
030      } else {
031        board[i] = FieldState.EMPTY;
032      }
033    }
034
035    return board;
036  }
037
038  @Test(dataProvider = "data")
039  public void checkIfMarkedProperIndexes(int lastHitIndex, FieldState[] board, Set<Integer> properIndices) {
040    SunkenShipMarker sunkenShipMarker = new SunkenShipMarker(board, lastHitIndex);
041
042    Set<Integer> testIndices = sunkenShipMarker.getIndexToColor();
043
044    assertTrue(testIndices.equals(properIndices));
045  }
046
047}
