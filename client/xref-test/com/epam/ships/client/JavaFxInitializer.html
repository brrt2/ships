3   import javafx.application.Application;
4   import javafx.stage.Stage;
5   import lombok.Getter;
6   
7   import java.awt.GraphicsEnvironment;
8   
9   /**
10   * Class using to initialize javafx thread to tests.
11   * @author Magdalena Aarsman
12   * @since 2018-01-01
13   */
14  public class JavaFxInitializer extends Application {
15  
16    private final static Object barrier = new Object();
17    @Getter
18    private static boolean launched = false;
19    @Getter
20    private static boolean enable = true;
21  
22    @Override
23    public void start(Stage primaryStage) throws Exception {
24      if (GraphicsEnvironment.isHeadless()) {
25        // non gui mode
26        enable = false;
27        return;
28      }
29      synchronized(barrier) {
30        barrier.notify();
31      }
32    }
33  
34    /**
35     * Launch javafx application.
36     */
37    public static void initialize() {
38      if (GraphicsEnvironment.isHeadless()) {
39        // non gui mode
40        enable = false;
41        return;
42      }
43      Thread t = new Thread("JavaFX Init Thread") {
44          public void run() {
45            launched = true;
46            Application.launch(JavaFxInitializer.class, new String[0]);
47          }
48        };
49        t.setDaemon(true);
50        t.start();
51        synchronized (barrier) {
52          try {
53            barrier.wait();
54          } catch (InterruptedException e) {
55            Thread.currentThread().interrupt();
56          }
57        }
58    }
59  }
