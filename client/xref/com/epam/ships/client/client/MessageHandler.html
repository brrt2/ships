1   package com.epam.ships.client.client;
2   
3   import com.epam.ships.infra.communication.api.Message;
4   import com.epam.ships.infra.communication.api.message.Header;
5   import com.epam.ships.infra.communication.api.message.Status;
6   import com.epam.ships.infra.logging.api.Target;
7   import com.epam.ships.infra.logging.core.SharedLogger;
8   import javafx.scene.control.Button;
9   import lombok.Getter;
10  
11  import java.util.Map;
12  
13  /**
14   * Handling message, enable to react in proper way to server messages.
15   *
16   * @author Magdalena Aarsman
17   * @since 2017-12-16
18   */
19  class MessageHandler {
20  
21    private static final Target logger = new SharedLogger(Client.class);
22    private final Map<Header, EventTrigger> triggers;
23    private Button eventButton;
24    @Getter
25    private boolean endConnectionTriggered;
26  
27    /**
28     * Creates an instance of MessageHandler and configures all possible triggers types.
29     */
30    MessageHandler(final Map<Header, EventTrigger> triggerMap) {
31      this.triggers = triggerMap;
32    }
33  
34    void setCurrentEventButton(Button eventButton) {
35      this.eventButton = eventButton;
36    }
37  
38    Button getCurrentEventButton() {
39      return eventButton;
40    }
41  
42    void handle(Message message) {
43      if (eventButton == null) {
44        throw new IllegalStateException("there is no object on which there can be fire event on");
45      }
46  
47      Header header = message.getHeader();
48  
49      if (!triggers.containsKey(header)) {
50        logger.error("message header: " + header + " is unknown");
51        return;
52      }
53      checkIfEndWillBeTriggered(message);
54      triggers.get(header).fire(eventButton, message.getStatement());
55    }
56  
57    private void checkIfEndWillBeTriggered(Message message) {
58      if (Status.END.equals(message.getStatus())) {
59        endConnectionTriggered = true;
60      }
61    }
62  }
