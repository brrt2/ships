1   package pl.korotkevics.ships.client.client;
2   
3   import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
4   
5   import java.util.EnumMap;
6   import java.util.Map;
7   
8   /**
9    * Helper class to build MessageHandler.
10   *
11   * @author Magdalena Aarsman
12   * @since 2018-01-02
13   */
14  
15  public class MessageHandlerBuilder {
16    private Map<Header, EventTrigger> triggers;
17  
18    /**
19     * Set triggers map to be enum map with Header as key.
20     *
21     * @return MessageHandlerBuilder
22     */
23    public MessageHandlerBuilder withEnumMap() {
24      triggers = new EnumMap<>(Header.class);
25      return this;
26    }
27  
28    /**
29     * Set default set of triggers and put it into map.
30     *
31     * @return MessageHandlerBuilder
32     */
33    public MessageHandlerBuilder withDefaultSetsOfTriggers() {
34      triggers.put(Header.OPPONENT_CONNECTED, new OpponentConnectedTrigger());
35      triggers.put(Header.SHOT, new OpponentShotTrigger());
36      triggers.put(Header.CONNECTION, new ConnectionEndTrigger());
37      triggers.put(Header.YOUR_TURN, new TurnTrigger());
38      triggers.put(Header.MISS, new MissShotTrigger());
39      triggers.put(Header.HIT, new HitShotTrigger());
40      triggers.put(Header.SHIP_DESTRUCTED, new HitShotTrigger());
41      triggers.put(Header.WIN, new WinTrigger());
42      triggers.put(Header.LOSE, new LoseTrigger());
43      triggers.put(Header.RANDOM_PLACEMENT, new RandomPlacementTrigger());
44      return this;
45    }
46  
47    /**
48     * Build MessageHandler.
49     *
50     * @return MessageHandler
51     */
52    public MessageHandler build() {
53      return new MessageHandler(triggers);
54    }
55  }
