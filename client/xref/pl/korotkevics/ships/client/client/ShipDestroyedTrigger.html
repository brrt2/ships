4   import javafx.scene.control.Button;

8   /**
9    * Trigger event reacting to destroyed ship.
10   *
11   * @author Magdalena Aarsman
12   * @since 2017-01-19
13   */
14  class ShipDestroyedTrigger implements EventTrigger {
15    @Override
16    public void fire(final Button button, final Message message) {
17      Platform.runLater(() -> button.fireEvent(new ShipDestroyedEvent()));
18    }
19    
20    @Override
21    public String provideDescription() {
22      return "What a luck! A ship has been destroyed. Go on!";
23    }
24  }
