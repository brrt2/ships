24  /**
25   * Main window (containing title and similar) controller.
26   *
27   * @author Magdalena Aarsman, Sandor Korotkevics
28   * @since 2017-12-14
29   */
30  
31  public class MainController {
32  
33    private static final Target logger = new SharedLogger(Client.class);
34  
35    private static final String DICTIONARY = "dict";
36  
37    @FXML
38    private Label gameTitle;
39  
40    @FXML
41    private Button russianButton;
42  
43    @FXML
44    private Pane mainPane;
45  
46    @Getter
47    private Client client;
48  
49    @FXML
50    private Button polishButton;
51  
52    @FXML
53    private Button englishButton;
54  
55    private ResourceBundle resourceBundle;
56  
57    /**
58     * Initialize main controller.
59     *
60     * @param client - Instance of Client, which communicate with server.
61     */
62    @FXML
63    public void initialize(final Client client) {
64      this.client = client;
65      this.loadDefaultView();
66  
67    }
68  
69    private void loadDefaultView() {
70      this.reloadView();
71    }
72  
73    private void reloadView() {
74      final FXMLLoader fxmlLoader = this.prepareFxmlLoader();
75      this.gameTitle.setText(this.resourceBundle.getString("gameTitle"));
76      try {
77        this.addToMainPane(fxmlLoader.load());
78      } catch (IOException e) {
79        logger.error(e.getMessage());
80      }
81    }
82  
83    private FXMLLoader prepareFxmlLoader() {
84      final String url = "/fxml/connectWindow.fxml";
85      final FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource(url));
86      this.resourceBundle = ResourceBundle.getBundle(DICTIONARY, new UTF8Control());
87      fxmlLoader.setResources(this.resourceBundle);
88      return fxmlLoader;
89    }
90  
91    private void addToMainPane(final Parent parent) {
92      this.mainPane.getChildren().clear();
93      this.mainPane.getChildren().add(parent);
94    }
95  
96    @FXML
97    void triggerPolishVersion(final ActionEvent event) {
98      logger.info("The language is set to " + OurLocale.POLISH);
99      Locale.setDefault(new Locale(OurLocale.POLISH.toString()));
100     this.reloadView();
101     this.changeTitleLabel();
102   }
103 
104   private void changeTitleLabel() {
105     Scene scene = mainPane.getScene();
106     Stage stage = (Stage) scene.getWindow();
107     stage.setTitle(this.resourceBundle.getString("gameTitle"));
108   }
109 
110   @FXML
111   void triggerEnglishVersion(final ActionEvent event) {
112     logger.info("The language is set to " + OurLocale.ENGLISH);
113     Locale.setDefault(new Locale(OurLocale.ENGLISH.toString()));
114     this.reloadView();
115     this.changeTitleLabel();
116   }
117 
118   @FXML
119   void triggerRussianVersion(final ActionEvent event) {
120     logger.info("The language is set to " + OurLocale.RUSSIAN);
121     Locale.setDefault(new Locale(OurLocale.RUSSIAN.toString()));
122     this.reloadView();
123     this.changeTitleLabel();
124   }
125 
126   void disableLocalizationButtons() {
127     this.hideButton(this.polishButton);
128     this.hideButton(this.englishButton);
129     this.hideButton(this.russianButton);
130   }
131 
132   private void hideButton(Button button) {
133     button.setDisable(true);
134     button.setVisible(false);
135   }
136 }
