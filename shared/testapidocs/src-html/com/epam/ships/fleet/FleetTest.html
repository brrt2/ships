001package com.epam.ships.fleet;
002
003import org.testng.annotations.Test;
004
005import java.util.Arrays;
006
007import static org.testng.Assert.assertEquals;
008import static org.testng.Assert.assertFalse;
009
010public class FleetTest {
011  @Test
012  public void itShouldCreateNonEmptyFleetWhenGivenWithShips() {
013    //given
014    Ship firstShip = Ship.ofMasts(Mast.ofIndex("3"), Mast.ofIndex("2"));
015    Ship secondShip = Ship.ofMasts(Mast.ofIndex("4"), Mast.ofIndex("1"));
016    //when
017    Fleet fleet = Fleet.ofShips(Arrays.asList(firstShip, secondShip));
018    //then
019    assertFalse(fleet.isDefeated());
020  }
021
022  @Test
023  public void shipIsDestructed() {
024    //given
025    Fleet fleet = Fleet.ofShips(Arrays.asList(Ship.ofMasts(Mast.ofIndex("3"), Mast.ofIndex("2"), Mast.ofIndex("1"))));
026    //when
027    Mast firstHit = Mast.ofIndex("3");
028    Mast secondHit = Mast.ofIndex("2");
029    Mast thirdHit = Mast.ofIndex("1");
030    //then
031    fleet.handleDamage(firstHit);
032    fleet.handleDamage(secondHit);
033    assertEquals(fleet.handleDamage(thirdHit), Damage.DESTRUCTED);
034  }
035
036  @Test
037  public void itIsMiss() {
038    //given
039    Fleet fleet = Fleet.ofShips(Arrays.asList(Ship.ofMasts(Mast.ofIndex("3"), Mast.ofIndex("2"), Mast.ofIndex("1"))));
040    //when
041    Mast miss = Mast.ofIndex("4");
042    //then
043    assertEquals(fleet.handleDamage(miss), Damage.MISSED);
044  }
045
046  @Test
047  public void stringRepresentationIsProper() {
048    //given - when
049    Fleet fleet = Fleet.ofShips(Arrays.asList(Ship.ofMasts(Mast.ofIndex("3"), Mast.ofIndex("2"), Mast.ofIndex("1"))));
050    //then
051    assertEquals(fleet.toString(), "Fleet(fleet={Mast(index=1)=Ship(masts=[Mast(index=1), Mast(index=2), Mast" +
052        "(index=3)]), Mast(index=2)=Ship(masts=[Mast(index=1), Mast(index=2), " +
053        "Mast(index=3)]), Mast(index=3)=Ship(masts=[Mast(index=1), Mast" +
054        "(index=2), Mast(index=3)])})");
055  }
056}
