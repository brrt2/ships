001package com.epam.ships.shared.infra.communication.core.json.io;
002
003import com.epam.ships.shared.infra.communication.api.Message;
004import com.epam.ships.shared.infra.communication.api.io.Sender;
005import com.epam.ships.shared.infra.communication.api.message.Author;
006import com.epam.ships.shared.infra.communication.api.message.Header;
007import com.epam.ships.shared.infra.communication.core.message.MessageBuilder;
008import org.testng.annotations.Test;
009
010import java.io.ByteArrayOutputStream;
011import java.io.OutputStream;
012
013import static org.testng.Assert.assertEquals;
014
015@Test(groups = {"integration"})
016public class JsonSenderTest {
017  public void itSendsProperlyEncodedMessage() {
018    //given
019    OutputStream outputStream = new ByteArrayOutputStream();
020    Sender sender = new JsonSender(outputStream);
021    Message message = new MessageBuilder()
022                          .withAuthor(Author.SERVER)
023                          .withHeader(Header.WIN)
024                          .build();
025    //when
026    sender.send(message);
027    //then
028    assertEquals(outputStream.toString(), "{\"header\":\"WIN\",\"status\":\"OK\"," +
029                                              "\"author\":\"SERVER\",\"statement\":\"\",\"fleet\":{\"fleet\":{}}}\n");
030  }
031}
