001package pl.korotkevics.ships.shared.infra.logging.core;
002
003import org.apache.logging.log4j.Level;
004import org.apache.logging.log4j.LogManager;
005import org.apache.logging.log4j.Logger;
006import pl.korotkevics.ships.shared.infra.logging.api.Target;
007
008/**
009 * Logger shared across modules. A Log4J2 wrapper.
010 * Log4J2 can be configured with log4j2.xml.
011 * A short manual is present within the XML file.
012 *
013 * @author Sandor Korotkevics
014 * @see Target
015 * @since 2017-12-10
016 */
017public class SharedLogger implements Target {
018  
019  final Level REPORT = Level.forName("REPORT", 50);
020  
021  private final Logger logger;
022  private final Class<?> clazz;
023  
024  /**
025   * @param clazz
026   *     class name to appear in log messages.
027   */
028  public SharedLogger(final Class<?> clazz) {
029    this.logger = LogManager.getLogger();
030    this.clazz = clazz;
031  }
032  
033  /**
034   * Used for logging normal (neutral) messages.
035   *
036   * @param message
037   *     to be logged.
038   *     Extra information will be
039   *     appended (such as class name,
040   *     datetime, level).
041   */
042  @Override
043  public void info(final Object message) {
044    this.logger.info(this.prepareDefaultMessage(message));
045  }
046  
047  /**
048   * Used for logging error messages
049   * (example: logging an exception
050   * message).
051   *
052   * @param message
053   *     to be logged.
054   *     Extra information will be
055   *     appended (such as class name,
056   *     datetime, level).
057   */
058  @Override
059  public void error(final Object message) {
060    this.logger.error(this.prepareDefaultMessage(message));
061  }
062  
063  /**
064   * Used for logging debug messages.
065   *
066   * @param message
067   *     to be logged.
068   *     Extra information will be
069   *     appended (such as class name,
070   *     datetime, level).
071   */
072  @Override
073  public void debug(final Object message) {
074    this.logger.debug(this.prepareDefaultMessage(message));
075  }
076  
077  
078  /**
079   * Used for logging game messages.
080   *
081   * @param message
082   *     to be logged.
083   * @param destinationLoggerName
084   *     an arbitrary logger name set with a configuration file.
085   */
086  @Override
087  public void report(final Object message, final String destinationLoggerName) {
088    this.logger.log(REPORT, this.prepareGameMessage(message, destinationLoggerName));
089  }
090  
091  private String prepareDefaultMessage(final Object message) {
092    return this.clazz.getSimpleName() + ": " + message;
093  }
094  
095  private String prepareGameMessage(final Object message, final String destinationLoggerName) {
096    return destinationLoggerName + ": " + message;
097  }
098}
