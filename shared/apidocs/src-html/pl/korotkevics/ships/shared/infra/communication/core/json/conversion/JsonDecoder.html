001package pl.korotkevics.ships.shared.infra.communication.core.json.conversion;
002
003import com.google.gson.Gson;
004import com.google.gson.GsonBuilder;
005import com.google.gson.JsonElement;
006
007import pl.korotkevics.ships.shared.infra.communication.api.Message;
008import pl.korotkevics.ships.shared.infra.communication.api.conversion.Decoder;
009import pl.korotkevics.ships.shared.infra.communication.core.message.BaseMessage;
010
011/**
012 * Converts a JsonElement into a BaseMessage instance.
013 *
014 * @author Sandor Korotkevics
015 * @see Decoder
016 * @see BaseMessage
017 * @since 2017-12-10
018 */
019public class JsonDecoder implements Decoder<JsonElement> {
020
021  /**
022   * Converts a JsonElement instance into a BaseMessage
023   * instance.
024   *
025   * @param jsonElement a JsonElement instance representing
026   *                    a BaseMessage.
027   * @return BaseMessage a result of conversion of JsonElement into BaseMessage
028   */
029  @Override
030  public Message decode(JsonElement jsonElement) {
031    Gson gson = new GsonBuilder().create();
032
033    return gson.fromJson(jsonElement, BaseMessage.class);
034  }
035}
