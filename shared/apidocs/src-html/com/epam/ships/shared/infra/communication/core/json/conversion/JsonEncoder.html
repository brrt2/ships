001package com.epam.ships.shared.infra.communication.core.json.conversion;
002
003import com.google.gson.Gson;
004import com.google.gson.GsonBuilder;
005import com.google.gson.JsonElement;
006
007import com.epam.ships.shared.infra.communication.api.Message;
008import com.epam.ships.shared.infra.communication.api.conversion.Encoder;
009import com.epam.ships.shared.infra.communication.core.message.BaseMessage;
010
011/**
012 * It converts a BaseMessage instance into a JsonElement.
013 * @author Sandor Korotkevics
014 * @see Encoder
015 * @see Message
016 * @since 2017-12-10
017 */
018public class JsonEncoder implements Encoder<JsonElement> {
019
020  /**
021   * It converts an instance of a class implementing
022   * BaseMessage interface into a JsonElement instance.
023   *
024   * @param message an instance of a class implementing
025   *                BaseMessage interface.
026   * @return a result of conversion of an instance of a class implementing
027   *     BaseMessage interface into a JsonElement instance.
028   */
029  @Override
030  public JsonElement encode(Message message) {
031    Gson gson = new GsonBuilder()
032        .enableComplexMapKeySerialization()
033        .create();
034    return gson.toJsonTree(message, BaseMessage.class);
035  }
036
037}
