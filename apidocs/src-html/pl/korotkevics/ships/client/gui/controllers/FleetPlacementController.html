142      event -> {

144          for (Node child : ((Group) event.getSource()).getChildren()) {

146            rec.setFill(Color.GRAY);

160        event.consume();

164  private void initializeBoard() {
165    final int margin = 50;
166    final NumberBinding allRectanglesWidth = Bindings.min(yourBoard.widthProperty(),
167        yourBoard.widthProperty().add(-margin));
168    final NumberBinding allRectanglesHeight = Bindings.min(yourBoard.heightProperty(),
169        yourBoard.widthProperty()).add(-margin);
170
171    for (int i = 0; i < BOARD_SIZE; i++) {
172      for (int j = 0; j < BOARD_SIZE; j++) {
173        final int recIndex = j * BOARD_SIZE + i;
174        final int fillIndex = i * BOARD_SIZE + j;
175
176        Rectangle yourRect = getYourRect(allRectanglesWidth, allRectanglesHeight);
177        yourBoard.add(yourRect, i, j);
178        setEventsOnField(yourRect, recIndex, fillIndex);
179        GridPane.setHalignment(yourRect, HPos.CENTER);
180      }
181    }
182  }
183
184  private Rectangle getYourRect(NumberBinding allRectanglesWidth,
185                                NumberBinding allRectanglesHeight) {
186    final int initialSize = 15;
187    Rectangle yourRect = new Rectangle(initialSize, initialSize, Color.GRAY);
188    yourRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
189    yourRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
190    return yourRect;
191  }

193  private void setEventsOnField(Rectangle rectangle, final int recIndex, final int fillIndex) {
194    setFieldOnDragEntered(rectangle, fillIndex);
195    rectangle.setOnDragOver(boardOnDragOver);
196    setFieldOnDragExited(rectangle, fillIndex);
197    setFieldOnDragDropped(rectangle, recIndex, fillIndex);
198  }

200  private void setOpacity(int index, int mastCount, double opacity) {
201    index += 1;
202
203    if (shipOrientation.equals(ShipOrientation.HORIZONTAL)) {
204      for (int i = 1; i < mastCount; i++) {
205        (yourBoard.getChildren().get(index + i * BOARD_SIZE)).setOpacity(opacity);
206      }
207    } else {
208      for (int i = 1; i < mastCount; i++) {
209        (yourBoard.getChildren().get(index + i)).setOpacity(opacity);
210      }
211    }
212  }
213
214  private void setFieldOnDragExited(Rectangle rectangle, int fillIndex) {
215    rectangle.setOnDragExited(
216        event -> {
217          double noOpacity = 1.0;
218          int mastCount = ((Group) event.getGestureSource()).getChildren().size();
219          if (isOutOfBound(fillIndex, mastCount)) {
220            return;
221          }
222          ((Rectangle) event.getSource()).setOpacity(noOpacity);
223          setOpacity(fillIndex, mastCount, noOpacity);
224          event.consume();
225        });
226  }
227
228  private void setFieldOnDragEntered(Rectangle rectangle, final int index) {
229    rectangle.setOnDragEntered(event -> {
230      double opacity = 0.5;
231      int mastCount = ((Group) event.getGestureSource()).getChildren().size();
232      if (shipOrientation.equals(ShipOrientation.HORIZONTAL) && isOutOfBound(index, mastCount)) {
233        return;
234      }
235      GridToBoardConverter gridToBoardConverter = new GridToBoardConverter(yourBoard);
236      ShipPlacementValidator shipPlacementValidator =
237          new ShipPlacementValidator(shipOrientation, index, mastCount,
238              gridToBoardConverter.convert());
239      if (!shipPlacementValidator.isPlacementValid()) {
240        return;
241      }
242
243      ((Rectangle) event.getSource()).setOpacity(opacity);
244      setOpacity(index, mastCount, opacity);
245      event.consume();
246    });
247  }
248
249  private boolean isOutOfBound(int index, int mastCount) {
250    final int notRectangleChildCount = 2;
251    if (shipOrientation.equals(ShipOrientation.HORIZONTAL)) {
252      return index + (mastCount - 1) * BOARD_SIZE
253          > yourBoard.getChildren().size() - notRectangleChildCount;
254    } else {
255      if (index + mastCount >= yourBoard.getChildren().size()) {
256        return true;
257      }
258    }
259    return false;
260  }
261
262  private void setFieldOnDragDropped(Rectangle rectangle, final int recIndex, final int index) {
263    rectangle.setOnDragDropped(event -> {
264      Dragboard db = event.getDragboard();
265      boolean success = false;
266      if (db.hasString()) {
267        success = true;
268      }
269      event.setDropCompleted(success);
270      int mastCount = ((Group) event.getGestureSource()).getChildren().size();
271      Mast[] masts = new Mast[mastCount];
272
273      if (shipOrientation.equals(ShipOrientation.HORIZONTAL) && isOutOfBound(index, mastCount)) {
274        shipPlacementSuccess = false;
275        return;
276      }
277
278      GridToBoardConverter gridToBoardConverter = new GridToBoardConverter(yourBoard);
279      ShipPlacementValidator shipPlacementValidator =
280          new ShipPlacementValidator(shipOrientation, index, mastCount,
281              gridToBoardConverter.convert());
282      if (!shipPlacementValidator.isPlacementValid()) {
283        shipPlacementSuccess = false;
284        return;
285      }
286      masts[0] = Mast.ofIndex(String.valueOf(recIndex));
287      rectangle.setFill(Color.GREEN);
288      placeShip(index, masts, mastCount, recIndex);
289      ships.add(Ship.ofMasts(masts));
290      if (ships.size() == SHIPS_COUNT) {
291        buttonReady.setDisable(false);
292        fleet = Fleet.ofShips(ships);
293      }
294      shipPlacementSuccess = true;
295      event.consume();
296    });
297  }
298
299  private void placeShip(int index, Mast[] masts, int mastCount, int recIndex) {
300    index += 1;
301
302    if (shipOrientation.equals(ShipOrientation.VERTICAL)) {
303      for (int i1 = 1; i1 < mastCount; i1++) {
304        ((Rectangle) yourBoard.getChildren().get(index + i1)).setFill(Color.GREEN);
305        masts[i1] = Mast.ofIndex(String.valueOf(recIndex + i1 * BOARD_SIZE));
306      }
307    } else {
308      for (int i1 = 1; i1 < mastCount; i1++) {
309        ((Rectangle) yourBoard.getChildren().get(index + i1 * BOARD_SIZE)).setFill(Color.GREEN);
310        masts[i1] = Mast.ofIndex(String.valueOf(recIndex + i1));
311      }
312    }
313  }

340  private void loadGameWindow() {
341    try {
342      this.getClient().sendFleet(fleet);
343
344      final String gameWindowUrl = "/fxml/gameWindow.fxml";
345      final FXMLLoader gameWindowLoader = new FXMLLoader(getClass().getResource(gameWindowUrl));
346      gameWindowLoader.setResources(this.resourceBundle);
347      final Parent gameWindow = gameWindowLoader.load();
348      final AnchorPane mainPane = (AnchorPane) mainAnchorPane.getParent();
349
350      Rectangle2D screenBounds = Screen.getPrimary().getVisualBounds();
351
352      Stage stage = (Stage) mainPane.getScene().getWindow();
353      final int shrinkSize = 300;
354      stage.setMinHeight(screenBounds.getHeight() - shrinkSize);
355      stage.setMinWidth(screenBounds.getWidth() - shrinkSize);
356
357      mainPane.getChildren().clear();
358      mainPane.getChildren().setAll(gameWindow);
359
360      GameWindowController gameController = gameWindowLoader.getController();
361      gameController.initializeClient();
362      gameController.initializeBoards(yourBoard);
363
364      final double margin = 0.0;
365      AnchorPane.setTopAnchor(gameWindow, margin);
366      AnchorPane.setBottomAnchor(gameWindow, margin);
367      AnchorPane.setLeftAnchor(gameWindow, margin);
368      AnchorPane.setRightAnchor(gameWindow, margin);
369
370    } catch (IOException e) {
371      logger.error(e.getMessage());
372    }
373  }

375  private void askForRandomFleet() {
376    final ButtonType yesButton = new ButtonType(this.resourceBundle.getString("yes"),
377        ButtonBar.ButtonData.YES);
378    final ButtonType noButton = new ButtonType(this.resourceBundle.getString("no"),
379        ButtonBar.ButtonData.NO);
380
381    Alert alert = showInfoAlert(yesButton, noButton);
382    if (alert.getResult() == noButton) {
383      return;
384    }
385
386    this.disableDragAndDropShips(true);
387    this.clearBoard();
388    this.buttonRandom.setDisable(true);
389    this.getClient().askForRandomFleet();
390  }
391
392  private Alert showInfoAlert(final ButtonType yesButton, final ButtonType noButton) {
393    final Alert alert = new Alert(Alert.AlertType.INFORMATION, this.resourceBundle
394        .getString("alertInfo")
395        + System.lineSeparator()
396        + System.lineSeparator()
397        + this.resourceBundle.getString("alertQuestion"),
398        yesButton, noButton);
399    alert.getDialogPane().setMinHeight(Region.USE_PREF_SIZE);
400    alert.setTitle(this.resourceBundle.getString("information"));
401    alert.setHeaderText(this.resourceBundle.getString("information"));
402    alert.showAndWait();
403    return alert;
404  }

406  private void disableDragAndDropShips(final boolean disable) {
407    this.disableDragAndDropForGroupOfShips(this.groupFourMastShips, disable);
408    this.disableDragAndDropForGroupOfShips(this.groupThreeMastShips, disable);
409    this.disableDragAndDropForGroupOfShips(this.groupTwoMastShips, disable);
410    this.disableDragAndDropForGroupOfShips(this.groupOneMastShips, disable);
411  }

413  private void disableDragAndDropForGroupOfShips(final Group shipsGroup, final boolean disable) {
414    for (Node ship : shipsGroup.getChildren()) {
415      ship.setDisable(disable);
416      for (Node child : ((Group) ship).getChildren()) {
417        final Rectangle rec = (Rectangle) child;
418        if (disable) {
419          rec.setFill(Color.GRAY);
420        } else {
421          final String purple = "#7A16C2";
422          rec.setFill(Color.web(purple));
423        }
424      }
425    }
426  }

428  private void getRandomFleet(final Fleet fleet) {
429    this.buttonReady.setDisable(false);
430    this.fleet = fleet;
431    this.drawFleet();
432    this.buttonRandom.setDisable(false);
433  }
434
435  private void drawFleet() {
436    this.fleet
437        .toIntegerList()
438        .forEach(i -> this.drawMast(this.convertToGridIndex(i)));
439  }
440
441  private void drawMast(int index) {
442    Rectangle rec = (Rectangle) (yourBoard.getChildren().get(index));
443    rec.setFill(Color.GREEN);
444  }

446  private int convertToGridIndex(final int index) {
447    final int column = index / BOARD_SIZE;
448    final int row = index - (column * BOARD_SIZE);
449    return row * BOARD_SIZE + column + 1;
450  }

452  private void clearBoardAction() {
453    buttonReady.setDisable(true);
454    this.disableDragAndDropShips(false);
455    clearBoard();
456  }

458  private void clearBoard() {
459    ships.clear();
460    for (int i = 1; i < 101; i++) {
461      ((Rectangle) yourBoard.getChildren().get(i)).setFill(Color.GRAY);
462    }
463  }

465  @Override
466  public void initialize(final URL location, final ResourceBundle resources) {
467    this.resourceBundle = resources;
468    buttonReady.setOnAction(event -> loadGameWindow());
469    buttonReady.setDisable(true);
470    initializeBoard();
471    addDragEventsToShips();
472    ships = new ArrayList<>(SHIPS_COUNT);
473    shipOrientation = ShipOrientation.VERTICAL;
474    rbVertical.setOnAction(event -> shipOrientation = ShipOrientation.VERTICAL);
475    rbHorizontal.setOnAction(event -> shipOrientation = ShipOrientation.HORIZONTAL);
476    buttonRandom.setOnAction(event -> askForRandomFleet());
477    eventButton.addEventHandler(RandomPlacementEvent.RANDOM_PLACEMENT_EVENT,
478        event -> getRandomFleet(event.getFleet()));
479    buttonClear.setOnAction(event -> clearBoardAction());
480    this.addDropShadows();
481  }
482
483  private void addDropShadows() {
484    final DropShadow shadow = new DropShadow();
485    this.buttonRandom.addEventHandler(MouseEvent.MOUSE_ENTERED, mouseEvent -> buttonRandom
486        .setEffect(shadow));
487    this.buttonRandom.addEventHandler(MouseEvent.MOUSE_EXITED, mouseEvent -> buttonRandom
488        .setEffect(null));
489    this.buttonReady.addEventHandler(MouseEvent.MOUSE_ENTERED, mouseEvent -> buttonReady
490        .setEffect(shadow));
491    this.buttonReady.addEventHandler(MouseEvent.MOUSE_EXITED, mouseEvent -> buttonReady
492        .setEffect(null));
493    this.buttonClear.addEventHandler(MouseEvent.MOUSE_ENTERED, mouseEvent -> buttonClear
494        .setEffect(shadow));
495    this.buttonClear.addEventHandler(MouseEvent.MOUSE_EXITED, mouseEvent -> buttonClear
496        .setEffect(null));
497  }
