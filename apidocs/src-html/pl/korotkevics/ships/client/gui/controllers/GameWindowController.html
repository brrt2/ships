046  private static final int BOARD_SIZE = 10;

074  private boolean hitShot;
075  private boolean shipDestroyed;

077  private void initializeTurn(final boolean myTurn) {
078    if (!myTurn) {
079      opponentBoard.setDisable(true);
080      final double opacity = 0.4;
081      opponentBoard.setOpacity(opacity);
082    }
083  }
084
085  void initializeClient() {
086    getClient().setEventTrigger(eventButton);
087  }
088
089  void initializeBoards(GridPane yourBoard) {
090    final int margin = 50;
091    final NumberBinding allRectanglesWidth = Bindings.min(this.yourBoard.heightProperty(),
092        this.yourBoard.widthProperty().add(-margin));
093    final NumberBinding allRectanglesHeight = Bindings.min(this.yourBoard.heightProperty(),
094        this.yourBoard.widthProperty()).add(-margin);
095
096    for (int i = 0; i < BOARD_SIZE; i++) {
097      for (int j = 0; j < BOARD_SIZE; j++) {
098        final int shotIndex = j * BOARD_SIZE + i;
099
100        Rectangle yourRect = getYourRect(allRectanglesWidth, allRectanglesHeight);
101        Rectangle opponentRect = getOpponentRectangle(allRectanglesWidth,
102            allRectanglesHeight, shotIndex);
103
104        this.yourBoard.add(yourRect, i, j);
105        opponentBoard.add(opponentRect, i, j);
106
107        GridPane.setHalignment(yourRect, HPos.CENTER);
108        GridPane.setHalignment(opponentRect, HPos.CENTER);
109      }
110    }
111
112    copyYourBoard(yourBoard);
113  }
114
115  private void copyYourBoard(GridPane yourBoard) {
116    for (int i = 1; i < yourBoard.getChildren().size(); i++) {
117      if (((Rectangle) yourBoard.getChildren().get(i)).getFill() == Color.GREEN) {
118        ((Rectangle) this.yourBoard.getChildren().get(i)).setFill(Color.GREEN);
119      }
120    }
121  }
122
123  private Client getClient() {
124    MainController mainController = (MainController) mainAnchorPane.getParent().getUserData();
125    return mainController.getClient();
126  }
127
128  private Rectangle getYourRect(NumberBinding allRectanglesWidth,
129                                NumberBinding allRectanglesHeight) {
130    final int initialSize = 15;
131    Rectangle yourRect = new Rectangle(initialSize, initialSize, Color.GRAY);
132    yourRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
133    yourRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
134
135    return yourRect;
136  }
137
138  private Rectangle getOpponentRectangle(NumberBinding allRectanglesWidth,
139                                         NumberBinding allRectanglesHeight,
140                                         final int shotIndex) {
141    final int initialSize = 15;
142    Rectangle opponentRect = new Rectangle(initialSize, initialSize, Color.GRAY);
143
144    opponentRect.setOnMouseClicked((MouseEvent mouseEvent) -> {
145      opponentBoard.setDisable(true);
146      if (opponentRect.getFill().equals(Color.GRAY)) {
147        opponentRect.setFill(Color.BLACK);
148      }
149      this.shotIndex = shotIndex;
150      getClient().sendShot(shotIndex);
151    });
152
153    opponentRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
154    opponentRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
155
156    return opponentRect;
157  }
158
159  private void opponentWithdraw() {
160    getClient().closeClient();
161    loadWithdrawScreen();
162  }
163
164  private void loadWithdrawScreen() {
165    try {
166      final String opponentWithdrawUrl = "/fxml/opponentWithdraw.fxml";
167      final FXMLLoader opponentWithdrawLoader =
168          new FXMLLoader(getClass().getResource(opponentWithdrawUrl));
169      opponentWithdrawLoader.setResources(this.resourceBundle);
170      final Parent opponentWithdraw = opponentWithdrawLoader.load();
171      final AnchorPane mainPane = (AnchorPane) mainAnchorPane.getParent();
172      final int sceneHeight = 400;
173      final int sceneWidth = 600;
174      final double margin = 0.0;
175
176      Stage stage = (Stage) mainPane.getScene().getWindow();
177      stage.setHeight(sceneHeight);
178      stage.setWidth(sceneWidth);
179
180      mainPane.getChildren().clear();
181      mainPane.getChildren().setAll(opponentWithdraw);
182
183      AnchorPane.setTopAnchor(opponentWithdraw, margin);
184      AnchorPane.setBottomAnchor(opponentWithdraw, margin);
185      AnchorPane.setLeftAnchor(opponentWithdraw, margin);
186      AnchorPane.setRightAnchor(opponentWithdraw, margin);
187
188    } catch (IOException e) {
189      logger.error(e.getMessage());
190    }
191  }
192
193  private void setOpponentShot(final int shotIndex) {
194    final int shotIndexInGrid = convertToGridIndex(shotIndex);
195    final Rectangle rec = (Rectangle) (yourBoard.getChildren().get(shotIndexInGrid));
196    if (rec.getFill() == Color.GREEN) {
197      rec.setFill(Color.RED);
198    } else if (rec.getFill().equals(Color.GRAY)){
199      rec.setFill(Color.BLACK);
200    }
201  }
202
203  private int convertToGridIndex(final int shotIndex) {
204    final int column = shotIndex / BOARD_SIZE;
205    final int row = shotIndex - (column * BOARD_SIZE);
206    return row * BOARD_SIZE + column + 1;
207  }
208
209  private void changeTurn() {
210    final double opacity = 0.4;
211    this.opponentBoard.setDisable(true);
212    this.opponentBoard.setOpacity(opacity);
213    this.infoLabel.setText(this.resourceBundle.getString("youMiss")
214                            + " "
215                            + this.resourceBundle.getString("opponentAction"));
216  }
217
218  private void markAsHit() {
219    final int shotIndexInGrid = this.convertToGridIndex(shotIndex);
220    final Rectangle rec = (Rectangle) (opponentBoard.getChildren().get(shotIndexInGrid));
221    rec.setFill(Color.RED);
222    this.hitShot = true;
223  }
224
225  private void setMyTurn() {
226    final double noOpacity = 1.0;
227    opponentBoard.setDisable(false);
228    opponentBoard.setOpacity(noOpacity);
229    this.infoLabel.setText(this.resourceBundle.getString("yourTurn"));
230    if(this.hitShot) {
231      this.infoLabel.setText(this.infoLabel.getText()
232          + " "
233          + this.resourceBundle.getString("youHit"));
234      this.hitShot = false;
235    } else if(this.shipDestroyed) {
236      this.infoLabel.setText(this.infoLabel.getText()
237          + " "
238          + this.resourceBundle.getString("shipDestroyed"));
239      this.shipDestroyed = false;
240    }
241  }
242  
243  @Override
244  public void initialize(final URL location, final ResourceBundle resources) {
245    this.resourceBundle = resources;
246    eventButton.addEventHandler(OpponentWithdrawEvent.OPPONENT_WITHDRAW,
247        opponentConnectedEvent -> opponentWithdraw());
248    eventButton.addEventHandler(OpponentShotEvent.OPPONENT_SHOT,
249        opponentShotEvent -> setOpponentShot(opponentShotEvent.getShotIndex()));
250    eventButton.addEventHandler(TurnChangeEvent.TURN_EVENT, event -> setMyTurn());
251    eventButton.addEventHandler(MissShotEvent.MISS_SHOT, event -> changeTurn());
252    eventButton.addEventHandler(HitShotEvent.HIT_SHOT, event -> markAsHit());
253    eventButton.addEventHandler(WinEvent.GAME_WIN, event -> renderAsWin());
254    eventButton.addEventHandler(LossEvent.GAME_LOSS, event -> renderAsLoss());
255    eventButton.addEventHandler(ShipDestroyedEvent.SHIP_DESTROYED, event -> this.shipDestroyed());
256    buttonEnd.setOnAction(actionEvent -> this.endTheGame());
257    infoLabel.setText(this.resourceBundle.getString("opponentAction"));
258    initializeTurn(false);
259  }
260
261  private void shipDestroyed() {
262    final int shotIndexInGrid = this.convertToGridIndex(shotIndex);
263    final Rectangle rec = (Rectangle) (opponentBoard.getChildren().get(shotIndexInGrid));
264    rec.setFill(Color.RED);
265    shipDestroyed = true;
266  }
267
268  private void endTheGame() {
269    Platform.exit();
270  }
271
272  private void renderAsWin() {
273    this.prepareToRenderAnyPossibleResult();
274    this.renderSpecificResult("youWin");
275    this.endStack.setVisible(true);
276    this.buttonEnd.setDisable(false);
277  }
278  
279  private void renderAsLoss() {
280    this.prepareToRenderAnyPossibleResult();
281    this.renderSpecificResult("youLose");
282    this.endStack.setVisible(true);
283    this.buttonEnd.setDisable(false);
284  }
285  
286  private void renderSpecificResult(String key) {
287    this.winLabel.setText(this.resourceBundle.getString(key));
288  }
289  
290  private void prepareToRenderAnyPossibleResult() {
291    this.disableWithdrawalPossibility();
292    this.disableBoards();
293    this.infoLabel.setText("");
294  }
295  
296  private void disableBoards() {
297    final double opacity = 0.4;
298    this.yourBoard.setDisable(true);
299    this.yourBoard.setOpacity(opacity);
300    this.opponentBoard.setDisable(true);
301    this.opponentBoard.setOpacity(opacity);
302    this.winLabel.setStyle("-fx-color: green");
303  }
304  
305  private void disableWithdrawalPossibility() {
306    this.eventButton.removeEventHandler(OpponentWithdrawEvent.OPPONENT_WITHDRAW,
307        opponentConnectedEvent -> opponentWithdraw());
308  }
