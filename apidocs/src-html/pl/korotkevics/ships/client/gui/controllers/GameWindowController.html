039public class GameWindowController implements Initializable {
040  
041  private static final Target logger = new SharedLogger(Client.class);
042
043  private static final int BOARD_SIZE = 10;
044
045  @FXML
046  private GridPane yourBoard;
047
048  @FXML
049  private GridPane opponentBoard;
050
051  @FXML
052  private AnchorPane mainAnchorPane;
053
054  @FXML
055  private Button eventButton;
056
057  @FXML
058  private Label winLabel;
059
060  private int shotIndex;
061  private ResourceBundle resourceBundle;
062  
063  private void initializeTurn(boolean myTurn) {
064    if (!myTurn) {
065      opponentBoard.setDisable(true);
066      final double opacity = 0.4;
067      opponentBoard.setOpacity(opacity);
068    }
069  }
070
071  void initializeClient() {
072    getClient().setEventTrigger(eventButton);
073  }
074
075  void initializeBoards(GridPane yourBoard) {
076    final int margin = 50;
077    final NumberBinding allRectanglesWidth = Bindings.min(this.yourBoard.heightProperty(),
078        this.yourBoard.widthProperty().add(-margin));
079    final NumberBinding allRectanglesHeight = Bindings.min(this.yourBoard.heightProperty(),
080        this.yourBoard.widthProperty()).add(-margin);
081
082    for (int i = 0; i < BOARD_SIZE; i++) {
083      for (int j = 0; j < BOARD_SIZE; j++) {
084        final int shotIndex = j * BOARD_SIZE + i;
085
086        Rectangle yourRect = getYourRect(allRectanglesWidth, allRectanglesHeight);
087        Rectangle opponentRect = getOpponentRectangle(allRectanglesWidth,
088            allRectanglesHeight, shotIndex);
089
090        this.yourBoard.add(yourRect, i, j);
091        opponentBoard.add(opponentRect, i, j);
092
093        GridPane.setHalignment(yourRect, HPos.CENTER);
094        GridPane.setHalignment(opponentRect, HPos.CENTER);
095      }
096    }
097
098    copyYourBoard(yourBoard);
099  }
100
101  private void copyYourBoard(GridPane yourBoard) {
102    for (int i = 1; i < yourBoard.getChildren().size(); i++) {
103      if (((Rectangle) yourBoard.getChildren().get(i)).getFill() == Color.GREEN) {
104        ((Rectangle) this.yourBoard.getChildren().get(i)).setFill(Color.GREEN);
105      }
106    }
107  }
108
109  private Client getClient() {
110    MainController mainController = (MainController) mainAnchorPane.getParent().getUserData();
111    return mainController.getClient();
112  }
113
114  private Rectangle getYourRect(NumberBinding allRectanglesWidth,
115                                NumberBinding allRectanglesHeight) {
116    final int initialSize = 15;
117    Rectangle yourRect = new Rectangle(initialSize, initialSize, Color.GRAY);
118    yourRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
119    yourRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
120
121    return yourRect;
122  }
123
124  private Rectangle getOpponentRectangle(NumberBinding allRectanglesWidth,
125                                         NumberBinding allRectanglesHeight,
126                                         final int shotIndex) {
127    final int initialSize = 15;
128    Rectangle opponentRect = new Rectangle(initialSize, initialSize, Color.GRAY);
129
130    opponentRect.setOnMouseClicked((MouseEvent mouseEvent) -> {
131      opponentBoard.setDisable(true);
132      if (opponentRect.getFill().equals(Color.GRAY)) {
133        opponentRect.setFill(Color.BLACK);
134      }
135      this.shotIndex = shotIndex;
136      getClient().sendShot(shotIndex);
137    });
138
139    opponentRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
140    opponentRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
141
142    return opponentRect;
143  }
144
145  private void opponentWithdraw() {
146    getClient().closeClient();
147    loadWithdrawScreen();
148  }
149
150  private void loadWithdrawScreen() {
151    try {
152      final String opponentWithdrawUrl = "/fxml/opponentWithdraw.fxml";
153      final FXMLLoader opponentWithdrawLoader =
154          new FXMLLoader(getClass().getResource(opponentWithdrawUrl));
155      opponentWithdrawLoader.setResources(this.resourceBundle);
156      final Parent opponentWithdraw = opponentWithdrawLoader.load();
157      final AnchorPane mainPane = (AnchorPane) mainAnchorPane.getParent();
158      final int sceneHeight = 400;
159      final int sceneWidth = 600;
160      final double margin = 0.0;
161
162      Stage stage = (Stage) mainPane.getScene().getWindow();
163      stage.setHeight(sceneHeight);
164      stage.setWidth(sceneWidth);
165
166      mainPane.getChildren().clear();
167      mainPane.getChildren().setAll(opponentWithdraw);
168
169      AnchorPane.setTopAnchor(opponentWithdraw, margin);
170      AnchorPane.setBottomAnchor(opponentWithdraw, margin);
171      AnchorPane.setLeftAnchor(opponentWithdraw, margin);
172      AnchorPane.setRightAnchor(opponentWithdraw, margin);
173
174    } catch (IOException e) {
175      logger.error(e.getMessage());
176    }
177  }
178
179  private void setOpponentShot(final int shotIndex) {
180    final int shotIndexInGrid = convertToGridIndex(shotIndex);
181    final Rectangle rec = (Rectangle) (yourBoard.getChildren().get(shotIndexInGrid));
182    if (rec.getFill() == Color.GREEN) {
183      rec.setFill(Color.RED);
184    } else if (rec.getFill().equals(Color.GRAY)){
185      rec.setFill(Color.BLACK);
186    }
187  }
188
189  private int convertToGridIndex(final int shotIndex) {
190    final int column = shotIndex / BOARD_SIZE;
191    final int row = shotIndex - (column * BOARD_SIZE);
192    return row * BOARD_SIZE + column + 1;
193  }
194
195  private void changeTurn() {
196    final double opacity = 0.4;
197    this.opponentBoard.setDisable(true);
198    this.opponentBoard.setOpacity(opacity);
199  }
200
201  private void markAsHit() {
202    final int shotIndexInGrid = this.convertToGridIndex(shotIndex);
203    final Rectangle rec = (Rectangle) (opponentBoard.getChildren().get(shotIndexInGrid));
204    rec.setFill(Color.RED);
205  }
206
207  private void setMyTurn() {
208    final double noOpacity = 1.0;
209    opponentBoard.setDisable(false);
210    opponentBoard.setOpacity(noOpacity);
211  }
212  
213  @Override
214  public void initialize(final URL location, final ResourceBundle resources) {
215    this.resourceBundle = resources;
216    eventButton.addEventHandler(OpponentWithdrawEvent.OPPONENT_WITHDRAW,
217        opponentConnectedEvent -> opponentWithdraw());
218    eventButton.addEventHandler(OpponentShotEvent.OPPONENT_SHOT,
219        opponentShotEvent -> setOpponentShot(opponentShotEvent.getShotIndex()));
220    eventButton.addEventHandler(TurnChangeEvent.TURN_EVENT, event -> setMyTurn());
221    eventButton.addEventHandler(MissShotEvent.MISS_SHOT, event -> changeTurn());
222    eventButton.addEventHandler(HitShotEvent.HIT_SHOT, event -> markAsHit());
223    eventButton.addEventHandler(WinEvent.GAME_WIN, event -> renderAsWin());
224    eventButton.addEventHandler(LooseEvent.GAME_LOSE, event -> renderAsLoss());
225  
226    initializeTurn(false);
227  }
228  
229  private void renderAsWin() {
230    this.prepareToRenderAnyPossibleResult();
231    this.renderSpecificResult("youWin");
232  }
233  
234  private void renderAsLoss() {
235    this.prepareToRenderAnyPossibleResult();
236    this.renderSpecificResult("youLose");
237  }
238  
239  private void renderSpecificResult(String key) {
240    this.winLabel.setText(this.resourceBundle.getString(key));
241  }
242  
243  private void prepareToRenderAnyPossibleResult() {
244    this.getClient().closeClient();
245    this.disableWithdrawalPossibility();
246    this.disableBoards();
247  }
248  
249  private void disableBoards() {
250    final double opacity = 0.4;
251    this.yourBoard.setDisable(true);
252    this.yourBoard.setOpacity(opacity);
253    this.opponentBoard.setDisable(true);
254    this.opponentBoard.setOpacity(opacity);
255    this.winLabel.setStyle("-fx-color: green");
256  }
257  
258  private void disableWithdrawalPossibility() {
259    this.eventButton.removeEventHandler(OpponentWithdrawEvent.OPPONENT_WITHDRAW,
260        opponentConnectedEvent -> opponentWithdraw());
261  }
