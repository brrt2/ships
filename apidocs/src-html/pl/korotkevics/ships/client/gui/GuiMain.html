001package pl.korotkevics.ships.client.gui;
002
003import javafx.application.Application;
004import javafx.application.Platform;
005import javafx.fxml.FXMLLoader;
006import javafx.scene.Parent;
007import javafx.scene.Scene;
008import javafx.stage.Stage;
009import javafx.stage.WindowEvent;
010import pl.korotkevics.ships.client.client.Client;
011import pl.korotkevics.ships.client.client.MessageHandlerBuilder;
012import pl.korotkevics.ships.client.gui.controllers.MainController;
013
014/**
015 * Main class of gui to initialize and start gui thread.
016 *
017 * @author Magdalena Aarsman
018 * @since 2017-12-17
019 */
020public class GuiMain extends Application {
021  @Override
022  public void start(Stage primaryStage) throws Exception {
023    Thread.currentThread().setName("gui - thread");
024    final String fxmlMainWindowPath = "/fxml/root.fxml";
025    final String windowTitle = "Battleships";
026    final boolean clientShouldRun = true;
027    final Client client = new Client(new MessageHandlerBuilder().withEnumMap()
028        .withDefaultSetsOfTriggers().withReporter().build(), clientShouldRun);
029    final int sceneWidth = 800;
030    final int sceneHeight = 400;
031
032    FXMLLoader rootLoader = new FXMLLoader(getClass().getResource(fxmlMainWindowPath));
033    Parent root = rootLoader.load();
034    MainController mainController = rootLoader.getController();
035    mainController.initialize(client);
036
037    primaryStage.setTitle(windowTitle);
038    primaryStage.setScene(new Scene(root, sceneWidth, sceneHeight));
039    primaryStage.setResizable(false);
040    primaryStage.show();
041
042    primaryStage.setOnHiding((WindowEvent event) -> {
043      client.closeClient();
044      Platform.exit();
045    });
046  }
047
048}
