001package pl.korotkevics.ships.client.localization;
002
003/**
004 * Locale representation.
005 * Implemented ourselves because JDK has just a few whereas Polish, Russian are not available.
006 *
007 * @author Sandor Korotkevics
008 * @since 2018-01-15
009 */
010
011public enum OurLocale {
012  ENGLISH {
013    @Override
014    public String toString() {
015      return "en";
016    }
017  },
018  POLISH {
019    @Override
020    public String toString() {
021      return "pl";
022    }
023  },
024  RUSSIAN {
025    @Override
026    public String toString() {
027      return "ru";
028    }
029  }
030}
