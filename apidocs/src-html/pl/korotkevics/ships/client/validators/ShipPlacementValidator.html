001package pl.korotkevics.ships.client.validators;
002
003import lombok.RequiredArgsConstructor;
004import pl.korotkevics.ships.client.gui.util.FieldState;
005import pl.korotkevics.ships.client.gui.util.ShipOrientation;
006
007import java.util.stream.IntStream;
008
009/**
010 * Enables validation of ships' placement.
011 *
012 * @author Magdalena Aarsman
013 * @since 2017-12-31
014 */
015@RequiredArgsConstructor
016public class ShipPlacementValidator {
017  private static final int BOARD_SIZE = 10;
018
019  private final ShipOrientation shipOrientation;
020  private final int shipStartIndex;
021  private final int mastCount;
022  private final FieldState[] board;
023
024  /**
025   * Check if ship placement is valid.
026   *
027   * @return true if it is valid, false if not.
028   */
029  public boolean isPlacementValid() {
030    return !this.isOutOfRange()
031        && (shipOrientation.equals(ShipOrientation.HORIZONTAL)
032        ? this.checkForHorizontal()
033        : this.checkForVertical());
034  }
035
036  private boolean checkIfBreakingThroughBorders() {
037    int startIndexInColumn = shipStartIndex % BOARD_SIZE;
038    return startIndexInColumn + (mastCount - 1) >= BOARD_SIZE;
039  }
040
041  private boolean checkForVertical() {
042    return !this.checkIfBreakingThroughBorders()
043        && !this.isShipOnShip()
044        && !this.isShipTouchOtherShip();
045  }
046
047  private boolean isOutOfRange() {
048    return shipOrientation.equals(ShipOrientation.VERTICAL)
049        ? shipStartIndex + mastCount > board.length
050        : shipStartIndex + (mastCount - 1) * BOARD_SIZE >= board.length;
051  }
052
053  private boolean isShipOnShip() {
054    return IntStream
055        .rangeClosed(shipStartIndex, shipStartIndex + mastCount - 1)
056        .anyMatch(p -> board[p].equals(FieldState.OCCUPIED));
057  }
058
059  private boolean isShipTouchOtherShip() {
060    return this.isShipAboveShip()
061        || this.isShipBelowShip()
062        || this.isShipOnTheRight()
063        || this.isShipOnTheLeft();
064  }
065
066  private boolean isShipAboveShip() {
067    final int checkWidthInCell = 3;
068    return !this.isInLastRow()
069        && IntStream.iterate(shipStartIndex - BOARD_SIZE, i -> i + BOARD_SIZE)
070        .limit(checkWidthInCell)
071        .filter(i -> i > 0)
072        .filter(i -> i + mastCount < board.length)
073        .anyMatch(i -> board[i + mastCount].equals(FieldState.OCCUPIED));
074  }
075
076  private boolean isInLastRow() {
077    final int lastRowModulo = 9;
078    return (shipStartIndex + (mastCount - 1)) % BOARD_SIZE == lastRowModulo;
079  }
080
081  private boolean isShipBelowShip() {
082    int checkWidthInCell = 3;
083    return !this.isInFirstRow() && shipStartIndex - 1 > 0
084        && IntStream.iterate(shipStartIndex - BOARD_SIZE, i -> i + BOARD_SIZE)
085        .limit(checkWidthInCell)
086        .filter(i -> i - 1 > 0)
087        .filter(i -> i < board.length)
088        .anyMatch(i -> board[i - 1].equals(FieldState.OCCUPIED));
089  }
090
091  private boolean isInFirstRow() {
092    return this.shipStartIndex % BOARD_SIZE == 0;
093  }
094
095  private boolean isShipOnTheRight() {
096    return IntStream.range(shipStartIndex, shipStartIndex + mastCount)
097        .filter(value -> value - BOARD_SIZE > 0)
098        .anyMatch(value -> board[value - BOARD_SIZE].equals(FieldState.OCCUPIED));
099  }
100
101  private boolean isShipOnTheLeft() {
102    return IntStream.range(shipStartIndex, shipStartIndex + mastCount)
103        .filter(value -> value > 0)
104        .filter(value -> value + BOARD_SIZE < board.length)
105        .anyMatch(value -> board[value + BOARD_SIZE].equals(FieldState.OCCUPIED));
106  }
107
108  private boolean checkForHorizontal() {
109    return !this.isShipOnShipHorizontal()
110        && !this.isShipTouchOtherShipHorizontal();
111  }
112
113  private boolean isShipTouchOtherShipHorizontal() {
114    return this.isShipBelowShipHorizontal()
115        || this.isShipAboveShipHorizontal()
116        || this.isShipOnTheRightHorizontal()
117        || this.isShipOnTheLeftHorizontal();
118  }
119
120  private boolean isShipOnShipHorizontal() {
121    return IntStream.iterate(shipStartIndex, n -> n + BOARD_SIZE)
122        .limit((mastCount - 1))
123        .anyMatch(n -> board[n].equals(FieldState.OCCUPIED));
124  }
125
126  private boolean isShipBelowShipHorizontal() {
127    return shipStartIndex % BOARD_SIZE != 9
128        && IntStream.iterate(shipStartIndex - BOARD_SIZE, n -> n + BOARD_SIZE)
129        .limit(mastCount + 2L)
130        .filter(n -> n > 0)
131        .filter(n -> n + 1 < board.length)
132        .anyMatch(n -> board[n + 1].equals(FieldState.OCCUPIED));
133  }
134
135  private boolean isShipAboveShipHorizontal() {
136    return !this.isInFirstRow()
137        && IntStream.iterate(shipStartIndex - BOARD_SIZE, n -> n + BOARD_SIZE)
138        .limit(mastCount + 2L)
139        .filter(n -> n > 0)
140        .filter(n -> n < board.length)
141        .anyMatch(n -> board[n - 1].equals(FieldState.OCCUPIED));
142  }
143
144  private boolean isShipOnTheRightHorizontal() {
145    return this.shipStartIndex + this.mastCount * BOARD_SIZE < this.board.length
146        && board[this.shipStartIndex + this.mastCount * BOARD_SIZE].equals(FieldState.OCCUPIED);
147  }
148
149  private boolean isShipOnTheLeftHorizontal() {
150    return this.shipStartIndex - BOARD_SIZE >= 1 && this.board[shipStartIndex - BOARD_SIZE]
151        .equals(FieldState.OCCUPIED);
152  }
153
154}
