001package pl.korotkevics.ships.shared.fleet;
002
003import lombok.AccessLevel;
004import lombok.AllArgsConstructor;
005import lombok.EqualsAndHashCode;
006import lombok.Getter;
007import lombok.ToString;
008
009import java.util.Arrays;
010import java.util.HashSet;
011import java.util.Set;
012
013/**
014 * Ship consisting of masts.
015 *
016 * @author Piotr Czy≈º, Sandor Korotkevics
017 * @see Mast
018 * @since 2017-12-19
019 */
020@EqualsAndHashCode
021@AllArgsConstructor(access = AccessLevel.PRIVATE)
022@ToString
023public class Ship {
024  @Getter(AccessLevel.PACKAGE)
025  private final Set<Mast> masts;
026
027  /**
028   * Static factory method.
029   *
030   * @param masts of ship
031   * @return Ship
032   */
033  public static Ship ofMasts(Mast... masts) {
034    return new Ship(new HashSet<>(Arrays.asList(masts)));
035  }
036}
