001package pl.korotkevics.ships.server;
002
003import pl.korotkevics.ships.shared.infra.communication.api.Message;
004import pl.korotkevics.ships.shared.infra.logging.api.Target;
005import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;
006
007import java.io.IOException;
008import java.net.Socket;
009import java.util.ArrayList;
010import java.util.List;
011
012/**
013 * It starts a server, gets its client sockets
014 * and adds them wrapped into own list.
015 *
016 * @author Piotr Czy≈º, Magdalena Aarsman, Sandor Korotkevics
017 * @since 2017-12-09
018 */
019public class CommunicationBus {
020
021  private static final int SERVER_PORT = 8189;
022  private final Target logger = new SharedLogger(CommunicationBus.class);
023  private final AppServer appServer;
024
025  private List<WrappedClient> clients;
026
027  /**
028   * Create communication bus instance.
029   *
030   * @throws IOException if an I/O error occurs when opening the socket.
031   */
032  public CommunicationBus() throws IOException {
033    appServer = new AppServer(SERVER_PORT);
034    clients = new ArrayList<>();
035  }
036
037  /**
038   * It ask AppServer for clients and then wrap them into wrapped client.
039   */
040  public void start() {
041    appServer.connectClients();
042    for (Socket socketClient : appServer.getClientSockets()) {
043      wrapClient(socketClient);
044    }
045    logger.info("Communication bus started..");
046  }
047
048  /**
049   * Close clients sockets and clear list of wrapped clients.
050   */
051  public void stop() {
052    for (WrappedClient c : clients) {
053      c.close();
054    }
055    clients.clear();
056  }
057
058  private void wrapClient(final Socket socketClient) {
059    WrappedClient wrappedClient = new WrappedClient(socketClient);
060    clients.add(wrappedClient);
061  }
062
063  /**
064   * It receives message from a player.
065   *
066   * @param sender player who sent a message
067   * @return received message
068   */
069  public Message receive(final WrappedClient sender) {
070    return sender.receive();
071  }
072
073  void send(final WrappedClient recipient, final Message response) {
074    recipient.send(response);
075  }
076
077  void sendToAll(final Message message) {
078    clients.forEach(wrappedClient -> send(wrappedClient, message));
079  }
080
081  /**
082   * Gets player which was connected earlier.
083   *
084   * @return WrappedClient
085   */
086  public WrappedClient getFirstClient() {
087    return this.clients.get(0);
088  }
089
090  /**
091   * Gets player which was connected as second one.
092   *
093   * @return WrappedClient
094   */
095  public WrappedClient getSecondClient() {
096    return this.clients.get(1);
097  }
098}
