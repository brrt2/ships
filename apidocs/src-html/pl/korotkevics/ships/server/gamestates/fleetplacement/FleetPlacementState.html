001package pl.korotkevics.ships.server.gamestates.fleetplacement;
002
003import pl.korotkevics.ships.server.CommunicationBus;
004import pl.korotkevics.ships.server.MessageSender;
005import pl.korotkevics.ships.server.TurnManager;
006import pl.korotkevics.ships.server.gamestates.GameState;
007import pl.korotkevics.ships.server.gamestates.endgame.GameEndWithWalkoverState;
008import pl.korotkevics.ships.server.gamestates.play.PlayState;
009import pl.korotkevics.ships.shared.fleet.Fleet;
010import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
011import pl.korotkevics.ships.shared.infra.logging.api.Target;
012import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;
013
014import java.util.ArrayList;
015import java.util.List;
016import java.util.Map;
017import java.util.concurrent.ConcurrentHashMap;
018
019/**
020 * Represents state of game in which players place their fleets.
021 *
022 * @author Piotr Czy≈º
023 * @since 2018-01-02
024 */
025public class FleetPlacementState implements GameState {
026  private final CommunicationBus communicationBus;
027  private final FleetPlacementProcessor fleetProcessor;
028  private final TurnManager turnManager;
029  private final Target logger = new SharedLogger(FleetPlacementState.class);
030  private final List<Fleet> fleets;
031
032  /**
033   * @param communicationBus client server communication bus
034   */
035  public FleetPlacementState(final CommunicationBus communicationBus) {
036    this.communicationBus = communicationBus;
037    this.turnManager = new TurnManager(communicationBus.getFirstClient(),
038        communicationBus.getSecondClient());
039    this.fleetProcessor = new FleetPlacementProcessor(communicationBus, new FleetGenerator());
040    fleets = new ArrayList<>();
041  }
042
043  /**
044   * Receives fleets from players, and transfers game to next state.
045   *
046   * @see PlayState
047   * @see GameEndWithWalkoverState
048   * @return GameState - PlayState
049   */
050  @Override
051  public GameState process() {
052    this.askPlayersForPlaceFleet();
053    this.placeFleet();
054    this.rest();
055    if (this.verifyClientsAreConnected()) {
056      return new PlayState(communicationBus, fleets);
057    }
058    return new GameEndWithWalkoverState(communicationBus);
059  }
060
061  private boolean verifyClientsAreConnected() {
062    return this.fleets.stream().noneMatch(Fleet::isDefeated);
063  }
064
065  private void placeFleet() {
066    final Map<Integer, Fleet> clientsFleet = new ConcurrentHashMap<>();
067    Thread firstClientFleetPlacement = new Thread(() ->
068        clientsFleet.put(0, this.fleetProcessor.placeFleet(this.turnManager.getCurrentPlayer())));
069    firstClientFleetPlacement.start();
070    Thread secondClientFleetPlacement = new Thread(() ->
071        clientsFleet.put(1, this.fleetProcessor.placeFleet(this.turnManager.getOtherPlayer())));
072    secondClientFleetPlacement.start();
073    try {
074      firstClientFleetPlacement.join();
075      secondClientFleetPlacement.join();
076    } catch (InterruptedException e) {
077      logger.error(e.getMessage());
078    }
079    this.fleets.add(clientsFleet.get(0));
080    this.fleets.add(clientsFleet.get(1));
081  }
082
083  private void askPlayersForPlaceFleet() {
084    MessageSender messageSender = new MessageSender(communicationBus, logger);
085    messageSender.sendToAll(Header.MANUAL_PLACEMENT);
086  }
087
088  private void rest() {
089    final long restTime = 300;
090    try {
091      Thread.sleep(restTime);
092    } catch (final InterruptedException e) {
093      logger.error(e.getMessage());
094      Thread.currentThread().interrupt();
095    }
096  }
097}
