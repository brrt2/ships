001package pl.korotkevics.ships.server.gamestates.endgame;
002
003import pl.korotkevics.ships.server.CommunicationBus;
004import pl.korotkevics.ships.server.MessageSender;
005import pl.korotkevics.ships.server.TurnManager;
006import pl.korotkevics.ships.server.gamestates.GameState;
007import pl.korotkevics.ships.shared.infra.communication.api.message.Header;
008import pl.korotkevics.ships.shared.infra.logging.api.Target;
009import pl.korotkevics.ships.shared.infra.logging.core.SharedLogger;
010
011/**
012 * Represents game state where game have a winner.
013 *
014 * @author Piotr Czy≈º
015 * @since 2018-01-02
016 */
017public class GameEndWithWinState implements GameState {
018  private final MessageSender messageSender;
019  private final Target logger = new SharedLogger(GameEndWithWinState.class);
020  private CommunicationBus communicationBus;
021  private TurnManager turnManager;
022
023  /**
024   * It creates state in which we have a winner.
025   *
026   * @param communicationBus client server communication bus
027   * @param turnManager      changes players' turns
028   */
029  public GameEndWithWinState(final CommunicationBus communicationBus,
030                             final TurnManager turnManager) {
031    this.communicationBus = communicationBus;
032    this.turnManager = turnManager;
033    messageSender = new MessageSender(communicationBus, logger);
034  }
035
036  /**
037   * Process end of game with the winner, and close communication bus.
038   *
039   * @return GameState
040   */
041  @Override
042  public GameState process() {
043    handleEndOfGame();
044    communicationBus.stop();
045    logger.info("Game ended");
046    return this;
047  }
048
049  /**
050   * Set that game should not by continued since we have a winner.
051   *
052   * @return boolean
053   */
054  @Override
055  public boolean shouldBeContinued() {
056    return false;
057  }
058
059  private void handleEndOfGame() {
060    this.messageSender.send(this.turnManager.getCurrentPlayer(), Header.WIN);
061    this.messageSender.send(this.turnManager.getOtherPlayer(), Header.LOSE);
062    this.rest();
063  }
064
065  private void rest() {
066    final int restTime = 100;
067    try {
068      Thread.sleep(restTime);
069    } catch (InterruptedException e) {
070      Thread.currentThread().interrupt();
071      logger.error(e.getMessage());
072    }
073  }
074}
