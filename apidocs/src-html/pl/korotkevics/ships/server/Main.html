001package pl.korotkevics.ships.server;
002
003import pl.korotkevics.ships.server.gamestates.GameState;
004import pl.korotkevics.ships.server.gamestates.connection.WaitingForPlayersState;
005
006import java.io.IOException;
007
008/**
009 * Main class of server application.
010 *
011 * @author Piotr Czy≈º
012 * @since 2017-12-09
013 */
014public class Main {
015  /**
016   * Main server method. It creates communication bus and starts the game.
017   * @param args is a conventional main method parameter.
018   * @throws IOException can be thrown due to establishing a socket connection.
019   */
020  public static void main(String[] args) throws IOException {
021    CommunicationBus communicationBus = new CommunicationBus();
022    
023    /*
024    Sonar reports an infinitive loop as a critical bug, and therefore
025    we could not think of a better workaround than presenting it as a
026    an effective infinitive loop (with true extracted to a flag).
027     */
028    boolean shouldRun = true;
029    while (shouldRun) {
030      GameState initialState = new WaitingForPlayersState(communicationBus);
031      new Game(initialState).play();
032    }
033  }
034}
