082  //Events Handlers
083  private EventHandler<MouseEvent> onMouseEnteredOnShip =
084      event -> {
085        for (Node child : ((Group) event.getSource()).getChildren()) {
086          Rectangle rec = (Rectangle) child;
087          rec.setFill(Color.BLUE);
088        }
089        event.consume();
090      };
091
092  private EventHandler<MouseEvent> onMouseExitFromShip =
093      event -> {
094        for (Node child : ((Group) event.getSource()).getChildren()) {
095          Rectangle rec = (Rectangle) child;
096          rec.setFill(Color.web("#7A16C2"));
097        }
098        event.consume();
099      };
100
101  private EventHandler<DragEvent> shipOnDragEntered =
102      event -> {
103        for (Node child : ((Group) event.getSource()).getChildren()) {
104          Rectangle rec = (Rectangle) child;
105          rec.setFill(Color.RED);
106        }
107        event.consume();
108      };
109
110  private EventHandler<MouseEvent> shipOnDragDetected =
111      event -> {
112        Dragboard db = ((Group) event.getSource()).startDragAndDrop(TransferMode.COPY);
113        ClipboardContent content = new ClipboardContent();
114        content.putString("Ship");
115        db.setContent(content);
116        event.consume();
117      };
118
119  private EventHandler<DragEvent> shipOnDragDone =
120      event -> {
121        if (shipPlacementSuccess) {
122          for (Node child : ((Group) event.getSource()).getChildren()) {
123            Rectangle rec = (Rectangle) child;
124            rec.setFill(Color.GRAY);
125          }
126          logger.info("end drag");
127          ((Group) event.getSource()).setDisable(true);
128        }
129        shipPlacementSuccess = false;
130        event.consume();
131      };
132
133  private EventHandler<DragEvent> boardOnDragOver =
134      event -> {
135        Dragboard db = event.getDragboard();
136        if (db.hasString()) {
137          event.acceptTransferModes(TransferMode.COPY);
138        }
139        event.consume();
140      };

142  @FXML
143  void initialize() {
144    buttonReady.setOnAction(event -> loadGameWindow());
145    buttonReady.setDisable(true);
146    initializeBoard();
147    addDragEventsToShips();
148    ships = new ArrayList<>(SHIPS_COUNT);
149    shipOrientation = Orientation.VERTICAL;
150    choiceBox.setItems(FXCollections.observableArrayList("VERTICAL", "HORIZONTAL"));
151    choiceBox.setValue("VERTICAL");
152    choiceBox.getSelectionModel()
153        .selectedItemProperty()
154        .addListener((observable, oldValue, newValue) -> {
155          shipOrientation = Orientation.valueOf((String) newValue);
156          logger.info((String) newValue);
157        });
158  }

160  private void initializeBoard() {
161    final int margin = 50;
162    final NumberBinding allRectanglesWidth = Bindings.min(yourBoard.widthProperty(),
163        yourBoard.widthProperty().add(-margin));
164    final NumberBinding allRectanglesHeight = Bindings.min(yourBoard.heightProperty(),
165        yourBoard.widthProperty()).add(-margin);
166
167    for (int i = 0; i < BOARD_SIZE; i++) {
168      for (int j = 0; j < BOARD_SIZE; j++) {
169        final int recIndex = j * BOARD_SIZE + i;
170        final int fillIndex = i * BOARD_SIZE + j;
171
172        Rectangle yourRect = getYourRect(allRectanglesWidth, allRectanglesHeight);
173        yourBoard.add(yourRect, i, j);
174        setEventsOnField(yourRect, recIndex, fillIndex);
175        GridPane.setHalignment(yourRect, HPos.CENTER);
176      }
177    }
178  }
179
180  private Rectangle getYourRect(NumberBinding allRectanglesWidth,
181                                NumberBinding allRectanglesHeight) {
182    final int initialSize = 15;
183    Rectangle yourRect = new Rectangle(initialSize, initialSize, Color.GRAY);
184    yourRect.widthProperty().bind(allRectanglesWidth.divide(BOARD_SIZE));
185    yourRect.heightProperty().bind(allRectanglesHeight.divide(BOARD_SIZE));
186    return yourRect;
187  }

189  private void setEventsOnField(Rectangle rectangle, final int recIndex, final int fillIndex) {
190    setFieldOnDragEntered(rectangle, fillIndex);
191    rectangle.setOnDragOver(boardOnDragOver);
192    rectangle.setOnDragExited(
193        event -> {
194          double noOpacity = 1.0;
195          int mastCount = ((Group) event.getGestureSource()).getChildren().size();
196          int index = fillIndex;
197          if (shipOrientation.equals(Orientation.VERTICAL)) {
198            index += 1;
199            ((Rectangle) event.getSource()).setOpacity(1);
200            for (int i = 1; i < mastCount; i++) {
201              ((Rectangle) yourBoard.getChildren().get(index + i)).setOpacity(noOpacity);
202            }
203          } else {
204            final int notRectangleChild = 2;
205            if (index + (mastCount - 1) * BOARD_SIZE
206                > yourBoard.getChildren().size() - notRectangleChild) {
207              return;
208            }
209            index += 1;
210            ((Rectangle) event.getSource()).setOpacity(1);
211            for (int i = 1; i < mastCount; i++) {
212              yourBoard.getChildren().get(index + i * BOARD_SIZE).setOpacity(noOpacity);
213            }
214          }
215          event.consume();
216        });
217    setFieldOnDragDropped(rectangle, recIndex, fillIndex);
218  }

220  private boolean checkIfNotSnaking(int index, int mastCount) {
221    int startIndexMod = index % BOARD_SIZE;
222    for (int i = 1; i < mastCount; i++) {
223      if ((index + i) % BOARD_SIZE < startIndexMod) {
224        return false;
225      }
226    }
227    return true;
228  }

230  private void setHorizontalOpacity(int index, int mastCount, double opacity) {
231    index += 1;
232    for (int i = 1; i < mastCount; i++) {
233      ((Rectangle) yourBoard.getChildren().get(index + i * BOARD_SIZE)).setOpacity(opacity);
234    }
235  }

237  private void setVerticalOpacity(int index, int mastCount, double opacity) {
238    index += 1;
239    for (int i = 1; i < mastCount; i++) {
240      ((Rectangle) yourBoard.getChildren().get(index + i)).setOpacity(opacity);
241    }
242  }

244  private boolean checkForVertical(int index, int mastCount) {
245    return isNotShipOnShip(index, mastCount)
246        && isNotShipAboveShip(index, mastCount)
247        && isNotShipBelowShip(index)
248        && isNotShipOnTheRight(index, mastCount)
249        && isNotShipOnTheLeft(index, mastCount);
250  }
251
252  private boolean isNotShipOnShip(int index, int mastCount) {
253    for (int i = 1; i < mastCount + 1; i++) {
254      if (index + i >= yourBoard.getChildren().size()) {
255        return false;
256      }
257
258      if (((Rectangle) yourBoard.getChildren().get(index + i)).getFill().equals(Color.GREEN)) {
259        return false;
260      }
261    }
262    return true;
263  }
264
265  private boolean isNotShipAboveShip(int index, int mastCount) {
266    if (index + mastCount + 1 < yourBoard.getChildren().size()) {
267      if (((Rectangle) yourBoard.getChildren().get(index + mastCount + 1)).getFill()
268          .equals(Color.GREEN)) {
269        return false;
270      }
271    }
272
273    return true;
274  }
275
276  private boolean isNotShipBelowShip(int index) {
277    if (index > 1 && index % BOARD_SIZE != 0) {
278      if (((Rectangle) yourBoard.getChildren().get(index)).getFill().equals(Color.GREEN)) {
279        return false;
280      }
281    }
282
283    return true;
284  }
285
286  private boolean isNotShipOnTheRight(int index, int mastCount) {
287    for (int i = 1; i < mastCount + 1; i++) {
288      if (index + i - BOARD_SIZE > 1) {
289        if (((Rectangle) yourBoard.getChildren().get(index + i - BOARD_SIZE)).getFill()
290            .equals(Color.GREEN)) {
291          return false;
292        }
293      }
294    }
295    return true;
296  }
297
298  private boolean isNotShipOnTheLeft(int index, int mastCount) {
299    for (int i = 1; i < mastCount + 1; i++) {
300      if (index + i + BOARD_SIZE < yourBoard.getChildren().size()) {
301        if (((Rectangle) yourBoard.getChildren().get(index + i + BOARD_SIZE)).getFill()
302            .equals(Color.GREEN)) {
303          return false;
304        }
305      }
306    }
307    return true;
308  }

310  private void setFieldOnDragEntered(Rectangle rectangle, final int index) {
311    rectangle.setOnDragEntered(event -> {
312      double opacity = 0.5;
313      int mastCount = ((Group) event.getGestureSource()).getChildren().size();
314      logger.info("index = " + index);
315      if (shipOrientation.equals(Orientation.VERTICAL)) {
316        if (!checkForVertical(index, mastCount) || !checkIfNotSnaking(index, mastCount)) {
317          return;
318        }
319        setVerticalOpacity(index, mastCount, opacity);
320        ((Rectangle) event.getSource()).setOpacity(opacity);
321      } else {
322        if (isOutOfBound(index, mastCount) || !checkForHorizontal(index, mastCount)) {
323          return;
324        }
325        ((Rectangle) event.getSource()).setOpacity(opacity);
326        setHorizontalOpacity(index, mastCount, opacity);
327      }
328      event.consume();
329    });
330  }

332  private boolean isNotShipOnShipHorizontal(int index, int mastCount) {
333    index += 1;
334    for (int i = 0; i < mastCount; i++) {
335      if (((Rectangle) yourBoard.getChildren().get(index + i * BOARD_SIZE)).getFill()
336          .equals(Color.GREEN)) {
337        return false;
338      }
339    }
340    return true;
341  }

343  private boolean isNotShipBelowShipHorizontal(int index, int mastCount) {
344    index += 1;
345    for (int i = 0; i < mastCount; i++) {
346      if ((index + i * BOARD_SIZE) % BOARD_SIZE == 0) {
347        return true;
348      }
349      final int oneRowBelow = 1;
350      if (index + oneRowBelow + i * BOARD_SIZE >= yourBoard.getChildren().size()) {
351        return true;
352      }
353
354      if (((Rectangle) yourBoard.getChildren().get(index + oneRowBelow + i * BOARD_SIZE)).getFill()
355          .equals(Color.GREEN)) {
356        return false;
357      }
358    }
359    return true;
360  }
361
362  private boolean isNotShipAboveShipHorizontal(int index, int mastCount) {
363    for (int i = 0; i < mastCount; i++) {
364      final int lastRow = 9;
365      if ((index + 1 + i * BOARD_SIZE) % BOARD_SIZE == lastRow) {
366        return true;
367      }
368      if (index + i * BOARD_SIZE >= yourBoard.getChildren().size() || index + i * BOARD_SIZE < 1) {
369        return true;
370      }
371
372      if (((Rectangle) yourBoard.getChildren().get(index + i * BOARD_SIZE)).getFill()
373          .equals(Color.GREEN)) {
374        return false;
375      }
376    }
377
378    return true;
379  }

381  private boolean isNotShipOnTheRightHorizontal(int index, int mastCount) {
382    index += 1;
383    if (index + mastCount * BOARD_SIZE >= yourBoard.getChildren().size()) {
384      return true;
385    }
386
387    if (((Rectangle) yourBoard.getChildren().get(index + mastCount * BOARD_SIZE)).getFill()
388        .equals(Color.GREEN)) {
389      return false;
390    }
391
392    return true;
393  }

395  private boolean isNotShipOnTheLeftHorizontal(int index) {
396    index += 1;
397    if (index - BOARD_SIZE < 1) {
398      return true;
399    }
400
401    if (((Rectangle) yourBoard.getChildren().get(index - BOARD_SIZE)).getFill()
402        .equals(Color.GREEN)) {
403      return false;
404    }

409  private boolean checkForHorizontal(int index, int mastCount) {
410    return isNotShipOnShipHorizontal(index, mastCount)
411        && isNotShipBelowShipHorizontal(index, mastCount)
412        && isNotShipAboveShipHorizontal(index, mastCount)
413        && isNotShipOnTheRightHorizontal(index, mastCount)
414        && isNotShipOnTheLeftHorizontal(index);
415  }
416
417  private boolean isOutOfBound(int index, int mastCount) {
418    final int notRectangleChildCount = 2;
419    if (index + (mastCount - 1) * BOARD_SIZE
420        > yourBoard.getChildren().size() - notRectangleChildCount) {
421      return true;
422    }
423
424    return false;
425  }

427  private void setFieldOnDragDropped(Rectangle rectangle, final int recIndex, final int fillIndex) {
428    rectangle.setOnDragDropped(event -> {
429      Dragboard db = event.getDragboard();
430      boolean success = false;
431      if (db.hasString()) {
432        success = true;
433      }
434      event.setDropCompleted(success);
435      int index = fillIndex;
436      int mastCount = ((Group) event.getGestureSource()).getChildren().size();
437      Mast[] masts = new Mast[mastCount];
438
439      if (shipOrientation.equals(Orientation.VERTICAL)) {
440        if (!checkForVertical(index, mastCount)) {
441          shipPlacementSuccess = false;
442          return;
443        }
444        if (!checkIfNotSnaking(index, mastCount)) {
445          shipPlacementSuccess = false;
446          return;
447        }
448        masts[0] = Mast.ofIndex(String.valueOf(recIndex));
449        rectangle.setFill(Color.GREEN);
450        index += 1;
451        for (int i1 = 1; i1 < mastCount; i1++) {
452          ((Rectangle) yourBoard.getChildren().get(index + i1)).setFill(Color.GREEN);
453          masts[i1] = Mast.ofIndex(String.valueOf(recIndex + i1 * BOARD_SIZE));
454        }
455      } else {
456        masts[0] = Mast.ofIndex(String.valueOf(recIndex));
457        if (index + (mastCount - 1) * BOARD_SIZE > yourBoard.getChildren().size() - 2) {
458          shipPlacementSuccess = false;
459          return;
460        }
461        index += 1;
462        rectangle.setFill(Color.GREEN);
463        for (int i1 = 1; i1 < mastCount; i1++) {
464          ((Rectangle) yourBoard.getChildren().get(index + i1 * BOARD_SIZE)).setFill(Color.GREEN);
465          masts[i1] = Mast.ofIndex(String.valueOf(recIndex + i1));
466        }
467      }
468      ships.add(Ship.ofMasts(masts));
469      if (ships.size() == SHIPS_COUNT) {
470        buttonReady.setDisable(false);
471      }
472      logger.info("first index of ship " + recIndex);
473      logger.info("ship has " + mastCount + " masts");
474      shipPlacementSuccess = true;
475      event.consume();
476    });
477  }

479  private void addDragEventsToShips() {
480    setEventsOnShips(groupFourMastShips);
481    setEventsOnShips(groupThreeMastShips);
482    setEventsOnShips(groupTwoMastShips);
483    setEventsOnShips(groupOneMastShips);
484  }

486  private void setEventsOnShips(Group shipsGroup) {
487    for (Node ship : shipsGroup.getChildren()) {
488      ship.setOnMouseEntered(onMouseEnteredOnShip);
489      ship.setOnMouseExited(onMouseExitFromShip);
490      ship.setOnDragEntered(shipOnDragEntered);
491      ship.setOnDragDetected(shipOnDragDetected);
492      ship.setOnDragDone(shipOnDragDone);
493    }
494  }

505  private void loadGameWindow() {
506    try {
507      getClient().sendFleet(Fleet.ofShips(ships));
508
509      final String gameWindowUrl = "/fxml/gameWindow.fxml";
510      final FXMLLoader gameWindowLoader = new FXMLLoader(getClass().getResource(gameWindowUrl));
511      final Parent gameWindow = gameWindowLoader.load();
512      final AnchorPane mainPane = (AnchorPane) mainAnchorPane.getParent();
513
514      Rectangle2D screenBounds = Screen.getPrimary().getVisualBounds();
515
516      Stage stage = (Stage) mainPane.getScene().getWindow();
517      final int shrinkSize = 300;
518      stage.setMinHeight(screenBounds.getHeight() - shrinkSize);
519      stage.setMinWidth(screenBounds.getWidth() - shrinkSize);
520
521      mainPane.getChildren().clear();
522      mainPane.getChildren().setAll(gameWindow);
523
524      GameController gameController = gameWindowLoader.getController();
525      gameController.initializeClient();
526      gameController.initializeBoards(yourBoard);
527
528      final double margin = 0.0;
529      AnchorPane.setTopAnchor(gameWindow, margin);
530      AnchorPane.setBottomAnchor(gameWindow, margin);
531      AnchorPane.setLeftAnchor(gameWindow, margin);
532      AnchorPane.setRightAnchor(gameWindow, margin);
533
534    } catch (IOException e) {
535      logger.error(e.getMessage());
536    }
537  }
