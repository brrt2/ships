001package com.epam.ships.client.gui.controllers;
002
003import com.epam.ships.client.client.Client;
004import com.epam.ships.client.gui.events.OpponentConnectedEvent;
005import com.epam.ships.client.validators.PortValidator;
006import com.epam.ships.shared.infra.logging.api.Target;
007import com.epam.ships.shared.infra.logging.core.SharedLogger;
008import javafx.fxml.FXML;
009import javafx.fxml.FXMLLoader;
010import javafx.geometry.Rectangle2D;
011import javafx.scene.Parent;
012import javafx.scene.control.Button;
013import javafx.scene.control.Label;
014import javafx.scene.control.TextField;
015import javafx.scene.image.ImageView;
016import javafx.scene.layout.AnchorPane;
017import javafx.scene.layout.GridPane;
018import javafx.scene.layout.Pane;
019import javafx.scene.layout.VBox;
020import javafx.stage.Screen;
021import javafx.stage.Stage;
022
023import java.io.IOException;
024import java.net.Socket;
025
026/**
027 * Connecting window controller.
028 * @author Magdalena Aarsman
029 * @since 2017-12-14
030 */
031public class StartWindowController {
032
033  private static final Target logger = new SharedLogger(Client.class);
034
035  private Client client;
036
037  @FXML
038  private GridPane gridPane;
039
040  @FXML
041  private ImageView imCannon;
042
043  @FXML
044  private VBox vbWheel;
045
046  @FXML
047  private TextField textFieldServerAddress;
048
049  @FXML
050  private TextField textFieldServerPort;
051
052  @FXML
053  private AnchorPane mainAnchorPane;
054
055  @FXML
056  private Label labelInvalidPort;
057
058  @FXML
059  private Button eventButton;
060
061  private PortValidator portValidator;
062
063  @FXML
064  void initialize() {
065    textFieldServerPort.textProperty().addListener((observableValue, s1, t1) -> {
066      if (!labelInvalidPort.getText().isEmpty()) {
067        labelInvalidPort.setText("");
068      }
069    });
070
071    portValidator = new PortValidator();
072    eventButton.addEventHandler(OpponentConnectedEvent.OPPONENT_CONNECTED,
073        event -> loadFleetPlacementWindow());
074
075    final String defaultHost = "127.0.0.1";
076    final String defaultPort = "8189";
077
078    textFieldServerAddress.setText(defaultHost);
079    textFieldServerPort.setText(defaultPort);
080  }
081
082  private void initializeClient() {
083    MainController mainController = (MainController) mainAnchorPane.getParent().getUserData();
084    this.client = mainController.getClient();
085    if (client == null) {
086      throw new IllegalStateException("client is not initialized!");
087    }
088
089    client.setEventTrigger(eventButton);
090  }
091
092  @FXML
093  private void onConnectPressed() {
094    try {
095      initializeClient();
096    } catch (IllegalStateException e) {
097      logger.error(e.getMessage());
098      return;
099    }
100
101    final String serverAddress = textFieldServerAddress.getText();
102    logger.info("server address: " + serverAddress);
103
104    int port;
105    try {
106      port = portValidator.asInt(textFieldServerPort.getText());
107    } catch (IllegalArgumentException e) {
108      labelInvalidPort.setText("invalid port number");
109      return;
110    }
111    logger.info("server port: " + port);
112
113    showLoadingWheel();
114    final boolean isConnected = client.connect(serverAddress, port, new Socket());
115
116    if (!isConnected) {
117      loadServerNotResponseView();
118    } else {
119      Thread clientThread = new Thread(client);
120      clientThread.start();
121    }
122  }
123
124  private void showLoadingWheel() {
125    final double opacity = 0.4;
126
127    gridPane.setDisable(true);
128    gridPane.setOpacity(opacity);
129    imCannon.setOpacity(opacity);
130    vbWheel.setVisible(true);
131  }
132
133  private void loadServerNotResponseView() {
134    try {
135      final String serverNotRespondingUrl = "/fxml/serverNotResponding.fxml";
136      final FXMLLoader notResponseLoader =
137          new FXMLLoader(getClass().getResource(serverNotRespondingUrl));
138      final Parent notResponse = notResponseLoader.load();
139      final Pane mainPane = (Pane) mainAnchorPane.getParent();
140      mainPane.getChildren().clear();
141      mainPane.getChildren().setAll(notResponse);
142    } catch (IOException e) {
143      logger.error(e.getMessage());
144    }
145  }
146
147  private void loadFleetPlacementWindow() {
148    try {
149      final String gameWindowUrl = "/fxml/fleetPlacement.fxml";
150      final FXMLLoader gameWindowLoader = new FXMLLoader(getClass().getResource(gameWindowUrl));
151      final Parent gameWindow = gameWindowLoader.load();
152      final AnchorPane mainPane = (AnchorPane) mainAnchorPane.getParent();
153
154      Rectangle2D screenBounds = Screen.getPrimary().getVisualBounds();
155
156      Stage stage = (Stage) mainPane.getScene().getWindow();
157      final int shrinkSize = 300;
158      stage.setMinHeight(screenBounds.getHeight() - shrinkSize);
159      stage.setMinWidth(screenBounds.getWidth() - shrinkSize);
160
161      mainPane.getChildren().clear();
162      mainPane.getChildren().setAll(gameWindow);
163
164      FleetPlacementController fleetPlacementController = gameWindowLoader.getController();
165      fleetPlacementController.initializeClient();
166
167      final double margin = 0.0;
168      AnchorPane.setTopAnchor(gameWindow, margin);
169      AnchorPane.setBottomAnchor(gameWindow, margin);
170      AnchorPane.setLeftAnchor(gameWindow, margin);
171      AnchorPane.setRightAnchor(gameWindow, margin);
172
173    } catch (IOException e) {
174      logger.error(e.getMessage());
175    }
176  }
177}
