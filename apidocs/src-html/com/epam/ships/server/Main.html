001package com.epam.ships.server;
002
003import com.epam.ships.infra.logging.api.Target;
004import com.epam.ships.infra.logging.core.SharedLogger;
005
006import java.io.IOException;
007
008/**
009 * Main class of server application.
010 *
011 * @author Piotr Czy≈º
012 * @since 2017-12-09
013 */
014public class Main {
015  private static final Target logger = new SharedLogger(Main.class);
016
017  /**
018   * Main server method. It creates communication bus and starts the game.
019   */
020  public static void main(String[] args) throws IOException {
021
022    CommunicationBus communicationBus = new CommunicationBus();
023    boolean shouldRun = true;
024    while (shouldRun) {
025      communicationBus.start();
026      final Game game = new Game(communicationBus);
027      game.play();
028      try {
029        Thread.sleep(300);
030      } catch (final InterruptedException e) {
031        logger.error(e.getMessage());
032        Thread.currentThread().interrupt();
033      }
034      communicationBus.stop();
035    }
036  }
037}
