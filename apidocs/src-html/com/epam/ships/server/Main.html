001package com.epam.ships.server;
002
003import com.epam.ships.infra.logging.api.Target;
004import com.epam.ships.infra.logging.core.SharedLogger;
005
006import java.io.IOException;
007
008public class Main {
009  private static final Target logger = new SharedLogger(Main.class);
010
011  public static void main(String[] args) throws IOException {
012
013    CommunicationBus communicationBus = new CommunicationBus();
014    boolean shouldRun = true;
015    while (shouldRun) {
016      communicationBus.start();
017      final Game game = new Game(communicationBus);
018      game.play();
019      try {
020        Thread.sleep(300);
021      } catch (final InterruptedException e) {
022        logger.error(e.getMessage());
023        Thread.currentThread().interrupt();
024      }
025      communicationBus.stop();
026    }
027  }
028}
