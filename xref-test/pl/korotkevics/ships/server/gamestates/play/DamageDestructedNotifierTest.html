17    public void shouldNotifyAboutDamage() {
18      //given
19      MessageSender messageSender = mock(MessageSender.class);
20      TurnManager turnManager = mock(TurnManager.class);
21      WrappedClient player = mock(WrappedClient.class);
22      WrappedClient opponent = mock(WrappedClient.class);
23      when(turnManager.getCurrentPlayer()).thenReturn(player);
24      when(turnManager.getOtherPlayer()).thenReturn(opponent);
25      Message message = new MessageBuilder()
26          .withAuthor(Author.SERVER)
27          .withHeader(Header.SHIP_DESTRUCTED)
28          .build();
29  
30      //when
31      new DamageDestructedNotifier(messageSender, turnManager).notify(message);
32  
33      //then
34      verify(messageSender, times(1)).send(player, Header.SHIP_DESTRUCTED);
35      verify(messageSender, times(1)).send(opponent, message);
36    }
