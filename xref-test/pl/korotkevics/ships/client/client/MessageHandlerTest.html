23  public class MessageHandlerTest {
24    
25    @BeforeClass
26    public void setUp() throws InterruptedException {
27      if (!JavaFxInitializer.isLaunched()) {
28        JavaFxInitializer.initialize();
29      }
30    }
31    
32    @BeforeMethod
33    public void skipTestIfThereIsNoGraphicsSupport() {
34      if (!JavaFxInitializer.isEnable()) {
35        throw new SkipException("skipping test because of lack of support for graphics");
36      }
37    }
38    
39    @Test
40    public void shouldCallConnectionEndTrigger() {
41      //given
42      ConnectionEndTrigger connectionEndTrigger = mock(ConnectionEndTrigger.class);
43      MessageHandler messageHandler = produceMessageHandler(connectionEndTrigger);
44      Message message = connectionEndMessage();
45      
46      //when
47      messageHandler.handle(message);
48      
49      //then
50      verify(connectionEndTrigger, times(1)).fire(messageHandler.getCurrentEventButton(), message);
51    }
52    
53    @Test
54    public void shouldReturnTrueAfterTriggerOpponentWithdrawEven() {
55      //given
56      ConnectionEndTrigger connectionEndTrigger = mock(ConnectionEndTrigger.class);
57      MessageHandler messageHandler = produceMessageHandler(connectionEndTrigger);
58      Message message = connectionEndMessage();
59      
60      //when
61      messageHandler.handle(message);
62      
63      assertTrue(messageHandler.isEndConnectionTriggered());
64    }
65    
66    private Message connectionEndMessage() {
67      return new MessageBuilder().withHeader(Header.CONNECTION).withStatus(Status.END)
68                 .withStatement("End of a message").build();
69    }
70    
71    private MessageHandler produceMessageHandler(EventTrigger eventTrigger) {
72      Map<Header, EventTrigger> triggers = new EnumMap<>(Header.class);
73      Reporter reporter = new Reporter("");
74      triggers.put(Header.CONNECTION, eventTrigger);
75      MessageHandler messageHandler = new MessageHandler(triggers, reporter);
76      messageHandler.setCurrentEventButton(new Button());
77      return messageHandler;
78    }
79  }
