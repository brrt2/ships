14  public class SunkenShipMarkerTest {
15    private static final int BOARD_SIZE = 10;
16  
17    @DataProvider
18    static Object[][] data() {
19      return new Object[][] {
20          {3, produceBoard(Arrays.asList(3,4,5)), new HashSet<Integer>(Arrays.asList(2,6,12,13,14,15,16))},
21          {3, produceBoard(Arrays.asList(3,13,23)), new HashSet<Integer>(Arrays.asList(2,12,22,32,33,34,4,14,24))}
22      };
23    }
24  
25    private static FieldState[] produceBoard(final List<Integer> occupiedPlaces) {
26      FieldState[] board = new FieldState[BOARD_SIZE * BOARD_SIZE];
27      for (int i = 0; i < BOARD_SIZE * BOARD_SIZE; i++) {
28        if (occupiedPlaces.contains(Integer.valueOf(i))) {
29          board[i] = FieldState.HIT;
30        } else {
31          board[i] = FieldState.EMPTY;
32        }
33      }
34  
35      return board;
36    }
37  
38    @Test(dataProvider = "data")
39    public void checkIfMarkedProperIndexes(int lastHitIndex, FieldState[] board, Set<Integer> properIndices) {
40      SunkenShipMarker sunkenShipMarker = new SunkenShipMarker(board, lastHitIndex);
41  
42      Set<Integer> testIndices = sunkenShipMarker.getIndexToColor();
43  
44      assertTrue(testIndices.equals(properIndices));
45    }

47  }
