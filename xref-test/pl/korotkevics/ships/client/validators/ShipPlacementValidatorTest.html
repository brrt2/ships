40    @Test(dataProvider = "validPlacement")
41    public void validPlacement(ShipOrientation shipOrientation, int shipStartIndex, int mastCount,
42                               FieldState[] board) {
43      //given
44      ShipPlacementValidator shipPlacementValidator =
45          new ShipPlacementValidator(shipOrientation, shipStartIndex, mastCount, board);

51    }

54    void invalidPlacement(ShipOrientation shipOrientation, int shipStartIndex, int mastCount,
55                          FieldState[] board) {
56      //given
57      ShipPlacementValidator shipPlacementValidator =
58          new ShipPlacementValidator(shipOrientation, shipStartIndex, mastCount, board);

65    private static FieldState [] producedBoard(List<Integer> occupiedPlaces) {
66      FieldState [] board = new FieldState[BOARD_SIZE * BOARD_SIZE];
67      for (int i = 0; i < BOARD_SIZE * BOARD_SIZE; i++) {
68        if(occupiedPlaces.contains(Integer.valueOf(i))) {
69          board[i] = FieldState.OCCUPIED;
70        } else {
71          board[i] = FieldState.EMPTY;
72        }
73      }
74  
75      return board;
76    }
