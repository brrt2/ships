16  public class JsonSenderTest {
17    public void itSendsProperlyEncodedMessage() {
18      //given
19      OutputStream outputStream = new ByteArrayOutputStream();
20      Sender sender = new JsonSender(outputStream);
21      Message message = new MessageBuilder()
22                            .withAuthor(Author.SERVER)
23                            .withHeader(Header.WIN)
24                            .build();
25      //when
26      sender.send(message);
27      //then
28      assertEquals(outputStream.toString(), "{\"header\":\"WIN\",\"status\":\"OK\"," +
29                                                "\"author\":\"SERVER\",\"statement\":\"\",\"fleet\":{\"fleet\":{}}}"+System.lineSeparator());
30    }
